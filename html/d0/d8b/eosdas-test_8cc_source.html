110  }
111 
112  if (parser_test)

115  if (scanner_test)

118  return (0);

121 void test_scanner()
122 {
123  int tok;
124 
125  cout << prompt << flush; // first prompt
126  while ((tok = hdfeoslex())) {
127  switch (tok) {
128  case GROUP:
129  cout << "GROUP" << endl;
130  break;
131  case END_GROUP:
132  cout << "END_GROUP" << endl;
133  break;
134  case OBJECT:
135  cout << "OBJECT" << endl;
136  break;
137  case END_OBJECT:
138  cout << "END_OBJECT" << endl;
139  break;
140 
141  case STR:
142  cout << "STR=" << hdfeoslval << endl;
143  break;
144  case INT:
145  cout << "INT=" << hdfeoslval << endl;
146  break;
147  case FLOAT:
148  cout << "FLOAT=" << hdfeoslval << endl;
149  break;
150 
151  case '=':
152  cout << "Equality" << endl;
153  break;
154  case '(':
155  cout << "LParen" << endl;
156  break;
157  case ')':
158  cout << "RParen" << endl;
159  break;
160  case ',':
161  cout << "Comma" << endl;
162  break;
163  case ';':
164  cout << "Semicolon" << endl;
165  break;
166 
167  default:
168  cout << "Error: Unrecognized input" << endl;
169  break;
170  }
171 
172  cout << prompt << flush; // print prompt after output
173  }
174 }
