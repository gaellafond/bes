72 void SQLTextContainer::comparatorSubst(SQL_CONSTRAINT_SET &where) {
73  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::comparatorSubstring start with size: " << where.size() << endl);
74  // for each constraint
75  SQL_CONSTRAINT_SET::iterator wi = where.begin();
76  while (wi != where.end()) {
81  string comparator = (*wi).getComparator();
82 
83  if (this->getOther(comparator)) {
84  // if comparator (as key) is found
85  // set its substitute
86  SQLConstraint c = (*wi);
91  c.setSubstitute(" " + comparator + " "); // to standard SQL
92  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::comparatorSubstring: " << c.toString() << endl);
93  where.erase(wi);
94  where.insert(c);
95  }
102  else if (comparator.compare("!=") == 0) {
103  // from standard dap
104  SQLConstraint c = (*wi);
105  c.setSubstitute("<>"); // to standard SQL
106  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::comparatorSubstring: " << c.toString() << endl);
107  where.erase(wi);
108  where.insert(c);
109  }
110  else if (comparator.compare("~=") == 0) {
111  // from standard dap
112  SQLConstraint c = (*wi);
113  c.setSubstitute(" LIKE "); // to standard SQL
114  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::comparatorSubstring: " << c.toString() << endl);
115  where.erase(wi);
116  where.insert(c);
117  }
118  wi++;
119  }
120  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::comparatorSubstring DONE" << endl);

123 string SQLTextContainer::buildQuery() {
124  BESDEBUG(SQL_NAME, "SQLTextContainer: Building SELECT query: " << endl);
125  SQLQuery &q = getQuery();
126 
132  string query = "SELECT ";
133  string buf;
134 
142  buf = BESUtil::www2id(this->get_constraint()); //onTheFly
147  size_t split = buf.find("&");
148 
149  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer split is: " << split << "\nBuf: " << buf << endl);
150 
151  string attributes;
152  string constraints;
153  if (split == 0) { // only constraints
154  constraints = buf;
155  }
156  else if (split < buf.size()) { // & is found in the middle
157  attributes = buf.substr(0, split);
158  constraints = buf.substr(split + 1);
159  }
160  else if (split >= buf.size()) {
165  attributes = buf;
166  }
167 
173  unique_ptr<SQL_ATTRIBUTE_SET_BYPOS> attrs(q.attrToSelect(attributes));
174  buf.clear();
175  if (!attrs->empty()) {
176  SQL_ATTRIBUTE_SET_BYPOS::iterator i = attrs->begin();
177  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer:: Attrs:" << (*i).getAttribute() << endl);
178  buf = (*i).getAttribute();
179  while (++i != attrs->end()) {
180  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer:: Attrs:" << (*i).getName() << endl);
181  buf += _SQLH_CONT_ATTR_JOIN + (*i).getAttribute();
182  }
183  }
184 
185  if (!buf.empty()) {
186  // use this as select clause
187  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer:: QUERY APPENDING:" << buf << endl);
188  query += buf;
189  buf.clear();
190  }
191  else
192  throw BESInternalFatalError("SQLTextContainer: Unable to build a correct SELECT clause", __FILE__, __LINE__);
193 
202  query += " FROM " + q.getFrom();
203 
218  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::WHERE starting with:" <<
219  "\nOnTheFly constraints ->" << constraints << "<-"
220  << endl);
221 
222  // where is optional and could be empty
223  if (!q.getWhere().empty()) {
228  this->comparatorSubst(q.getWhere());
229 
230  SQLQuery::whereIterator wi = q.getWhere().begin();
231  buf += (*wi++).toString();
232  while (wi != q.getWhere().end()) {
233  buf += _SQLH_CONT_REG_CONSTR_BASE_JOIN;
234  buf += (*wi).toString();
235  wi++;
236  }
237  }
238 
239  // ON THE FLY SPECIFIED CONSTRAINTS
240 
241  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::constrToWhere mid: " << buf << endl);
242  if (!constraints.empty()) {
243 
244  // on the fly constraints are not empty
245  SQL_CONSTRAINT_SET in_constr = SQLQuery::loadConstraints(constraints);
246 
251  this->comparatorSubst(in_constr);
252  // on the fly constraints are correctly loaded into 'in_constr'
253  if (!in_constr.empty()) {
254  SQLQuery::whereIterator wi = in_constr.begin();
255  do {
256  if (!buf.empty()) {
257  // '_where' contain previous conditions join (AND)
258  buf += _SQLH_CONT_REG_CONSTR_BASE_JOIN;
259  }
269  SQLQuery::attrIterator ai = q.getSelect().find((*wi).getAttribute());
270 
271  // if found
272  if (ai != q.getSelect().end()) {
301  buf += (*ai).getFullName();
302  }
303  // append comparator
304  if (!(*wi).getSubstitute().empty())
305  buf += (*wi).getSubstitute();
306  else
307  buf += (*wi).getComparator();
308  // append value
309  buf += (*wi).getVal();
310  } while (++wi != in_constr.end());
311  }
312  }
313 
314  if (!buf.empty()) {
315  // we have: simply use it
316  query += " WHERE " + buf;
317  }
318 
319  query += ";";
320 
325  BESDEBUG(SQL_NAME, "SQLTextContainer: QUERY: " << query << endl);
326  return query;

486 bool SQLTextContainer::read() throw(BESError) {
487  BESDEBUG(SQL_NAME, "SQLTextContainer: start reading dataset" << endl);
488  // to read the accessed file
489  std::fstream file;
490  // FIXME Move the compiles inside the try/catch
491  regex_t _reg_v;
492  if (regcomp(&_reg_v, _SQLH_DATASET_REG_VAR, REG_EXTENDED) != 0) {
493  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
494  }
495  regex_t _reg_c;
496  if (regcomp(&_reg_c, _SQLH_DATASET_REG_COMMENTS, REG_EXTENDED) != 0) {
497  regfree(&_reg_v);
498  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
499  }
500  regex_t _reg_t;
501  if (regcomp(&_reg_t, _SQLH_DATASET_REG_TAGS, REG_EXTENDED) != 0) {
502  regfree(&_reg_v);
503  regfree(&_reg_c);
504  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
505  }
506  regex_t _reg_k;
507  if (regcomp(&_reg_k, _SQLH_DATASET_REG_KEY_VAL, REG_EXTENDED) != 0) {
508  regfree(&_reg_v);
509  regfree(&_reg_c);
510  regfree(&_reg_t);
511  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
512  }
513  regex_t _reg_dv;
514  if (regcomp(&_reg_dv, _SQLH_DATASET_REG_DEFVAR, REG_EXTENDED) != 0) {
515  regfree(&_reg_v);
516  regfree(&_reg_c);
517  regfree(&_reg_t);
518  regfree(&_reg_k);
519  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
520  }
521 
522  try {
523  regex_t _reg_v, _reg_c, _reg_t, _reg_k, _reg_d;
524  if (regcomp(&_reg_v, _SQLH_DATASET_REG_VAR, REG_EXTENDED) != 0
525  || regcomp(&_reg_c, _SQLH_DATASET_REG_COMMENTS, REG_EXTENDED) != 0
526  || regcomp(&_reg_t, _SQLH_DATASET_REG_TAGS, REG_EXTENDED) != 0
527  || regcomp(&_reg_k, _SQLH_DATASET_REG_KEY_VAL, REG_EXTENDED) != 0
528  || regcomp(&_reg_dv, _SQLH_DATASET_REG_DEFVAR, REG_EXTENDED) != 0) {
529  throw BESInternalError("SQLTextContainer: Unable to compile the REGEX", __FILE__, __LINE__);
530  }
531 
532  // Open the dataset file.
533  file.open(get_real_name().c_str(), std::fstream::in);
534 
535  if (file.fail() || !(file.is_open()))
536  throw BESInternalError("Cannot open accessed file!", __FILE__, __LINE__);
537 
538  // This variable represents minimum dataset requirements.
539  // TODO when 'api=name' what is required might change
540  std::bitset<_SQLH_DATASET_PARTS_NUM> required = requirements();
541  // initialize completion diary registering where we were
542  std::bitset<_SQLH_DATASET_PARTS_NUM> complete;
543  // index to the (actual) reading section part (where we are)
544  _SQLH_DATASET_PARTS reading_part; //no need to set
545  // initial condition
546  bool _start = true;
547 
548  SQLH_DATASET_SECTION new_section;
549 
550  // map used to perform variable substitutions into the dataset
551  std::map<string, string> vars;
552 
553  // READ THE DATASET
554  while (!file.eof()) {
555  std::string row; // line buffer
556  getline(file, row);
557  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: READING ROW->" << row << "<-" << endl);
558 
559  if (!row.empty()) {
560  // index of matched regex groups
561  regmatch_t indexes[_SQLH_DATASET_MAX_REG_GROUPS];
562  // search for COMMENTS
563  if (regexec(&_reg_c, row.c_str(), (size_t) _SQLH_DATASET_REG_COMMENTS_GROUPS, indexes, 0) == 0) {
564  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: Comment->" << row << "<-" << endl);
565  }
566 
567  // variable definition
568  else if (regexec(&_reg_dv, row.c_str(), (size_t) _SQLH_DATASET_REG_DEFVAR_GROUPS, indexes, 0) == 0) {
569  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: defineVariable->" << row << "<-" << endl);
570  defineVariable(vars, row, indexes[_SQLH_DATASET_REG_DEFVAR_KEY_GROUP],
571  indexes[_SQLH_DATASET_REG_DEFVAR_VAL_GROUP]);
572  }
573 
574  // TAGS (which delimit parts)
575  else if (regexec(&_reg_t, row.c_str(), (size_t) _SQLH_DATASET_REG_TAGS_GROUPS, indexes, 0) == 0) {
576  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: partTracking->" << row << "<-" << endl);
577  partTracking(_start, new_section, required, complete, reading_part, row,
578  indexes[_SQLH_DATASET_REG_TAGS_GROUP]);
579  }
580 
581  // scan for all other lines
582  else {
583  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: variableSubstitution->" << row << "<-" << endl);
584  // variable substitution; substitute $NAME$ everywhere on this row.
585  while (regexec(&_reg_v, row.c_str(), (size_t) _SQLH_DATASET_REG_VAR_GROUPS, indexes, 0) == 0) {
586  variableSubstitution(vars, row, indexes[_SQLH_DATASET_REG_VAR_GROUP]);
587  }
588 
589  // KEY=VALUE lines
590  if (regexec(&_reg_k, row.c_str(), (size_t) _SQLH_DATASET_REG_KEY_VAL_GROUPS, indexes, 0) == 0) {
591  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: addKeyValue->" << row << "<-" << endl);
592  addKeyValue(new_section, required, complete, reading_part, row,
593  indexes[_SQLH_DATASET_REG_KEY_GROUP], indexes[_SQLH_DATASET_REG_VAL_GROUP]);
594  }
595 
596  // all other lines (free form line)
597  else {
598  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: addFreeForm->" << row << "<-" << endl);
599  addFreeForm(
600  //section to add to
601  new_section,
602  //actual section completion diary.
603  complete, reading_part, row);
604  }
605  }
606  } // if row!empty
607  }
608 
609  if (file.is_open())
610  file.close();
611 
612  regfree(&_reg_v);
613  regfree(&_reg_c);
614  regfree(&_reg_t);
615  regfree(&_reg_k);
616  regfree(&_reg_dv);
617 
618  if (_start) {
619  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: empty dataset" << endl);
620  // empty dataset?
621  return false;
622  }
623  else {
624  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: last addNewSection" << endl);
625  addNewSection(new_section, required, complete);
626  return true;
627  } //final if
628  } //try
629  catch (BESError &e) {
630  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: caught exception: " << e.get_message() << endl);
631 
632  regfree(&_reg_v);
633  regfree(&_reg_c);
634  regfree(&_reg_t);
635  regfree(&_reg_k);
636  regfree(&_reg_dv);
637 
638  throw;
639  } //catch
640  catch (...) {
641  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: caught unknown exception" << endl);
642 
643  regfree(&_reg_v);
644  regfree(&_reg_c);
645  regfree(&_reg_t);
646  regfree(&_reg_k);
647  regfree(&_reg_dv);
648 
649  throw;
650  } //catch
651 } //end

672 void SQLTextContainer::partTracking(bool &start, SQLH_DATASET_SECTION &new_section,
673  std::bitset<_SQLH_DATASET_PARTS_NUM> &required,
674  std::bitset<_SQLH_DATASET_PARTS_NUM> &complete,
675  _SQLH_DATASET_PARTS &reading_part, string &row,
676  regmatch_t &tag) {
677  BESDEBUG(SQL_NAME_TEST, "Found a tag. row: " << row << " Tag start: "
678  << tag.rm_so << "Tag end: " << tag.rm_eo << endl);
679 
680  // SECTION TAG
681  if (row.compare(tag.rm_so, tag.rm_eo - tag.rm_so, _SQLH_DATASET_SECTION_TAG) == 0) {
687  if (start) //starting condition
688  start = false;
689  else {
690  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer: addNewSection->" << row << "<-" << endl);
694  addNewSection(new_section, required, complete);
695 
696  // reset completion status
697  complete.reset();
698  // clear section
699  new_section = SQLH_DATASET_SECTION();
700  }
701 
702  reading_part = section_tag; // where we are
703 
704  BESDEBUG(SQL_NAME, "SQLTextContainer: Start reading a [section]:" << endl);
705  }
706  // SELECT TAG
707  else if (row.compare(tag.rm_so, tag.rm_eo - tag.rm_so, _SQLH_DATASET_SELECT_TAG) == 0) {
708  reading_part = select_tag;
709  BESDEBUG(SQL_NAME, "Entering a [select] part: " << endl);
710  }
711  // FROM TAG
712  else if (row.compare(tag.rm_so, tag.rm_eo - tag.rm_so, _SQLH_DATASET_FROM_TAG) == 0) {
713  reading_part = from_tag;
714  BESDEBUG(SQL_NAME, "Entering a [from] part: " << endl);
715  }
716  // WHERE TAG
717  else if (row.compare(tag.rm_so, tag.rm_eo - tag.rm_so, _SQLH_DATASET_WHERE_TAG) == 0) {
718  reading_part = where_tag;
719  BESDEBUG(SQL_NAME, "Entering a [where] part: " << endl);
720  }
721  // OTHER TAG
722  else if (row.compare(tag.rm_so, tag.rm_eo - tag.rm_so, _SQLH_DATASET_OTHER_TAG) == 0) {
723  reading_part = other_tag;
724  BESDEBUG(SQL_NAME, "Entering a [other] part: " << endl);
725  }
726 }

738 void SQLTextContainer::addNewSection(SQLH_DATASET_SECTION &new_section,
739  std::bitset<_SQLH_DATASET_PARTS_NUM> &required,
740  std::bitset<_SQLH_DATASET_PARTS_NUM> &complete) {
741  if (complete.test(where_tag)) {
742  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::addNewSection() Processing WHERE tag" << endl);
743  // load constraints into the query
744  SQL_CONSTRAINT_SET cs = SQLQuery::loadConstraints(_where);
745  new_section.query.setWhere(cs);
746  //clear _where buffer
747  _where.clear();
748  // if where is still empty
749  if (new_section.query.getWhere().empty())
750  complete.set(where_tag, false);
751  }
752 
753  if (complete.test(select_tag)) {
754  BESDEBUG(SQL_NAME_TEST, "SQLTextContainer::addNewSection() Processing SELECT tag." << endl);
761  SQL_ATTRIBUTE_SET_BYNAME as = SQLQuery::loadAttributes(_attribute);
762  new_section.query.setSelect(as);
763  //clear _attribute buffer
764  _attribute.clear();
765  // if where is still empty
766  if (new_section.query.getSelect().empty())
767  complete.set(select_tag, false);
768  }
769 
770  if (required == (required & complete)) {
771  // completion match requirements
772  BESDEBUG(SQL_NAME_TEST,
773  "SQLTextContainer::addNewSection() push_back complete section->" << new_section.api << "<-" << endl);
774  // push a copy of the last parsed section
775  _dataset->push_back(new_section);
776  }
777  else {
778  BESDEBUG(SQL_NAME_TEST,
779  "SQLTextContainer: Unable to complete dataset scan. "
780  "\nNOTE: could be BigEndian or LittleEndian:" <<
781  "\nCOMPLETITION MAP: " << complete.to_string() <<
782  "\nREQUIRED MAP: " << required.to_string() <<
783  "\nINDEX: \nsection_tag=0\napi=1\nserver=2\nport=3\ndbname=4"
784  "\nuser=5\npass=6\nselect_tag=7\nfrom_tag=8\nwhere_tag=9"
785  "\nother_tag=10" << endl);
786  throw BESError("SQL Handler: Unable to complete dataset scan. Section has missing elements.",
787  BES_SYNTAX_USER_ERROR, __FILE__, __LINE__);
788  }
