51 SQLContainerFactory::getContainer(BESDataHandlerInterface &dhi) {
52  if (dhi.container) {
53  if (cs) {
54  BESDEBUG(SQL_NAME, "SQLContainerFactory: SQL storage container found. " << endl);
55 
56  SQLContainer *c;
57 #if 1
58 
61  if ((c = dynamic_cast<SQLContainer *>(dhi.container))) {
62  if (c->isReady()) {
63  BESDEBUG(SQL_NAME,
64  "SQLContainerFactory: Container is already a READY SQLContainer" << endl);
65  //this is a copy
66  return c;//c->ptr_duplicate();
67  }
68  else {
69  BESDEBUG(SQL_NAME,
70  "SQLContainerFactory: Container is already SQLContainer but not READY" << endl);
75  if (!c->setup()) {
76  BESDEBUG(SQL_NAME, "ERROR: SQLRequestHandler" << endl);
77  throw BESInternalError(
78  "SQLContainerFactory: Failed to rebuild sql dataset",
79  __FILE__, __LINE__);
80  }
84  cs->update(c->ptr_duplicate());
85  //return the same copy for local (to the caller) use
86  return c;
87  }
88  }
89  else {
90 #endif
91  BESDEBUG(SQL_NAME, "SQLContainerFactory: Searching for cached container." << endl);
92  // NOTE: using getName
93  c = cs->look_for(getName(*dhi.container));
94  if (c) {
95  BESDEBUG(SQL_NAME, "SQLContainerFactory: Container is cached" << endl);
101  c->set_attributes(dhi.container->get_attributes());
102  c->set_constraint(dhi.container->get_constraint());
103 
104  //return the same copy for local (to the caller) use
105  return c;
106  }
107  else {
108  BESDEBUG(SQL_NAME, "SQLContainerFactory: Container not cached: Try to build it" << endl);
109  try {
110  BESDEBUG(SQL_NAME, "SQLContainerFactory: Setup SQLContainer: Creating dataset." << endl);
116  SQLContainer *c = buildContainer(*dhi.container);
117 
123  cs->add_container(c->ptr_duplicate());
124 
125  //return the same copy for local (to the caller) use
126  return c;
127  }
128  catch (std::exception &e) {
129  BESDEBUG(SQL_NAME, "SQLContainerFactory: Setup SQLContainer: ERROR->" << e.what());
130  throw BESInternalError(e.what(), __FILE__, __LINE__);
131  }
132  } //end dhi.container is not an SQLContainer
133  } // endif (bc)
134 
135  } //endif (cs)
136  else {
137  BESDEBUG(SQL_NAME, "SQLContainerFactory: Not using persistence" << endl);
138  //this is a copy
139  SQLContainer *c = buildContainer(*dhi.container);
140 
141  //return the same copy for local (to the caller) use
142  return c;
143  }
144  }
145  else // dhi.container is NULL
146  return NULL;
147 }
