142  static _JOIN *doActions(SQLActionFactory<CODE_TYPE, ARGS_TYPE, OUT_TYPE> &_af, MERGE merge_f = &merge, JOIN join_f =
143  &join, bool multiple_args = true) {
144  if (!merge_f) merge_f = &merge;
145  if (!join_f) join_f = &join;
146 
147  CODE_TYPE *code = NULL;
148  _JOIN *_join = NULL;
149  // do actions
150  do {
151  _MERGE *_merge = NULL;
152 
153  // get next code
154  if ((code = _af.getCode())) {
155  // do actions for that code
156  SQLActionList<ARGS_TYPE, OUT_TYPE> &actions = _af.getActions(code);
157 
158  // Used to initialize first time 'args'
159  bool first_time = true;
165  try {
170  ARGS_TYPE *args = NULL;
171  while (actions.hasNext()) {
172  // multiple_args || first_time == true?
173  if (multiple_args)
174  args = _af.getArgs(code);
175  else if (first_time) {
176  first_time = false;
177  args = _af.getArgs(code);
178  }
179 
180  // tryNext action and merge results
181  _merge = merge_f(_merge, tryNext(actions, args));
182  }
183  }
184  catch (SQLInternalError &ie) {
185  BESDEBUG(SQL_NAME, "SQLInternalError handled" << std::endl);
186  }
187  } // if code
188  else
189  // ERROR_TYPE *code==NULL
190  break;
191 
192  // join resulting _merge_status
193  _join = join_f(_join, _merge);
194 
195  // resetting _merge_status
196  _merge = NULL;
197 
198  // while NOT stop
199  } while (!_af.stop(code)); //(!stop(code));
200 
201  return _join;
202  } // doActions
