252  template<size_t sz>
253  static std::list<matched<sz> > match(const char *regex, std::bitset<sz> &groups, std::string &row) {
254  if (!regex) {
255  throw BESInternalFatalError("StringMatch::match Unable to use NULL regex pointer",
256  __FILE__, __LINE__);
257  }
258  // regex
259  regex_t _reg_expr;
260  // compile regex
261  if (regcomp(&_reg_expr, regex, REG_EXTENDED) != 0) {
262  throw BESInternalError("StringMatch: Unable to compile the REGEX: " + (std::string) regex,
263  __FILE__, __LINE__);
264  }
265  // returned
266  std::list<matched<sz> > ret;
267 
271  regmatch_t indexes[sz]; // index matches
272 
273  try {
274  // while regex match into the string
275  while (regexec(&_reg_expr, row.c_str(), groups.size(), indexes, 0) == 0) {
276  // this substring matching groups to add to ret
277  std::bitset<sz> gr;
278  // each matching substring
279  gr.reset();
285  //for (size_t i = groups._Find_next(0); i < sz; i = groups._Find_next(i)) {
286  for (size_t i = get_next(groups, 0); i < sz; i = get_next(groups, ++i)) {
287  BESDEBUG(SQL_NAME, "Processing group is " << i << " of " << sz << std::endl);
288  // check if limits are acceptable (group is found)
289  if (indexes[i].rm_so != -1) { // if group is found
290  BESDEBUG(SQL_NAME, "Group number " << i << " is found:" <<
291  " Substring-> start: " << indexes[i].rm_so <<
292  " Substring-> size: " << indexes[i].rm_eo - indexes[i].rm_so
293  <<
294  " String-> size: " << row.size() << std::endl);
295  gr.set(i, true);
296  }
297 
298  BESDEBUG(SQL_NAME, "Done with group " << i << std::endl);
299  }
300  // if some group is found in the substring
301  if (gr.any()) {
302  BESDEBUG(SQL_NAME, "StringMatch::Some group found, extracting: " << std::endl);
307  size_t i_matched = 0;
308  // build the matched object
309  matched<sz> m(gr);
310  //for (size_t i = m.getMap().FIND_FIRST; i < sz; i = m.getMap()._Find_next(i)) {
311  for (size_t i = get_next(m.getMap(), 0); i < sz; i = get_next(m.getMap(), ++i)) {
312  m.setMatch(i_matched, row.substr(indexes[i].rm_so, indexes[i].rm_eo - indexes[i].rm_so));
313  BESDEBUG(SQL_NAME, "StringMatch::match extracting: " <<
314  "group: " << i << " found: "
315  << m.getMatch(i_matched) << std::endl);
316  i_matched++;
317  }
318  // store found groups
319  ret.push_back(m);
320  }
321 
322  // delete matching string (entire string)
323  row.erase(indexes[0].rm_so, indexes[0].rm_eo - indexes[0].rm_so);
324 
325  BESDEBUG(SQL_NAME, "StringMatch::match remain: " << row << std::endl);
326 
327  }
328  regfree(&_reg_expr);
329  }
330  catch (...) {
331  regfree(&_reg_expr);
332  throw;
333  }
334  BESDEBUG(SQL_NAME, "StringMatch::match done" << std::endl);
335  return ret;
336  }
