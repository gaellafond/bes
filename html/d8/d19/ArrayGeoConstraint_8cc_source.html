91 bool ArrayGeoConstraint::build_lat_lon_maps()
92 {
93  // Find the longitude dimension: Assume it is the rightmost
94  set_longitude_rightmost(true);
95  set_lon_dim(d_array->dim_begin() + (d_array->dimensions() - 1));
96 
97  int number_elements_longitude = d_array->dimension_size(get_lon_dim());
98  double *lon_map = new double[number_elements_longitude];
99  for (int i = 0; i < number_elements_longitude; ++i) {
100  lon_map[i] = ((d_extent.d_right - d_extent.d_left) / (number_elements_longitude - 1)) * i + d_extent.d_left;
101  }
102  set_lon(lon_map);
103  set_lon_length(number_elements_longitude);
104 
105  // Find the latitude dimension: Assume it is the next-rightmost
106  set_lat_dim(d_array->dim_begin() + (d_array->dimensions() - 2));
107 
108  int number_elements_latitude = d_array->dimension_size(get_lat_dim());
109  double *lat_map = new double[number_elements_latitude];
110  for (int i = 0; i < number_elements_latitude; ++i) {
111  lat_map[i] = ((d_extent.d_bottom - d_extent.d_top) / (number_elements_latitude - 1)) * i + d_extent.d_top;
112  }
113  set_lat(lat_map);
114  set_lat_length(number_elements_latitude);
115 
116  return get_lat() && get_lon();
117 }

146 void ArrayGeoConstraint::apply_constraint_to_data()
147 {
148  if (!is_bounding_box_set())
149  throw InternalErr(
150  "The Latitude and Longitude constraints must be set before calling\n\
151  apply_constraint_to_data().");
152 
153  if (get_latitude_sense() == inverted) {
154  int tmp = get_latitude_index_top();
155  set_latitude_index_top(get_latitude_index_bottom());
156  set_latitude_index_bottom(tmp);
157  }
158 
159  // It's easy to flip the Latitude values; if the bottom index value
160  // is before/above the top index, return an error explaining that.
161  if (get_latitude_index_top() > get_latitude_index_bottom())
162  throw Error("The upper and lower latitude indexes appear to be reversed. Please provide\nthe latitude bounding box numbers giving the northern-most latitude first.");
163 
164  d_array->add_constraint(get_lat_dim(),
165  get_latitude_index_top(), 1,
166  get_latitude_index_bottom());
167 
168  // Does the longitude constraint cross the edge of the longitude vector?
169  // If so, reorder the data (array).
170  if (get_longitude_index_left() > get_longitude_index_right()) {
171  reorder_data_longitude_axis(*d_array, get_lon_dim());
172 
173  // Now the data are all in local storage
174 
175  // alter the indexes; the left index has now been moved to 0, and the right
176  // index is now at lon_vector_length-left+right.
177  set_longitude_index_right(get_lon_length() - get_longitude_index_left()
178  + get_longitude_index_right());
179  set_longitude_index_left(0);
180  }
181  // If the constraint used the -180/179 (neg_pos) notation, transform
182  // the longitude map s it uses the -180/179 notation. Note that at this
183  // point, d_longitude always uses the pos notation because of the earlier
184  // conditional transformation.
185 
186  // Apply constraint; stride is always one
187  d_array->add_constraint(get_lon_dim(),
188  get_longitude_index_left(), 1,
189  get_longitude_index_right());
190 
191  // Load the array if it has been read, which will be the case if
192  // reorder_data_longitude_axis() has been called.
193  if (get_array_data()) {
194 
195  int size = d_array->val2buf(get_array_data());
196 
197  if (size != get_array_data_size())
198  throw InternalErr
199  ("Expected data size not copied to the Grid's buffer.");
200  d_array->set_read_p(true);
201  }
202  else {
203  d_array->read();
204  }
205 }
