189 string EffectiveUrlCache::get_effective_url(const string &source_url)
190 {
191 
192  // This lock is a RAII implementation. It will block until the mutex is
193  // available and the lock will be released when the instance is destroyed.
194  std::lock_guard<std::mutex> lock_me(d_cache_lock_mutex);
195 
196  BESDEBUG(MODULE, prolog << "BEGIN url: " << source_url << endl);
197  string effective_url_str = source_url;
198 
199  if(is_enabled()){
200 
201  BESDEBUG(MODULE_DUMPER, prolog << "dump: " << endl << dump() << endl);
202 
203  size_t match_length=0;
204 
205  // if it's not an HTTP url there is nothing to cache.
206  if (source_url.find("http://") != 0 && source_url.find("https://") != 0) {
207  BESDEBUG(MODULE, prolog << "END Not an HTTP request, SKIPPING." << endl);
208  return effective_url_str;
209  }
210 
211  BESRegex *skip_regex = get_skip_regex();
212  if( skip_regex ) {
213  match_length = skip_regex->match(source_url.c_str(), source_url.length());
214  if (match_length == source_url.length()) {
215  BESDEBUG(MODULE, prolog << "END Candidate url matches the "
216  "no_redirects_regex_pattern [" << skip_regex->pattern() <<
217  "][match_length=" << match_length << "] SKIPPING." << endl);
218  return effective_url_str;
219  }
220  BESDEBUG(MODULE, prolog << "Candidate url: '" << source_url << "' does NOT match the "
221  "skip_regex pattern [" << skip_regex->pattern() << "]" << endl);
222  }
223  else {
224  BESDEBUG(MODULE, prolog << "The cache_effective_urls_skip_regex() was NOT SET "<< endl);
225  }
226 
227  http::EffectiveUrl *effective_url = get(source_url);
228 
229  // See if the data_access_url has already been processed into a terminal URL
230  bool retrieve_and_cache = !effective_url; // If there's no effective_url we gotta go get it.
231  if(effective_url){
232  BESDEBUG(MODULE, prolog << "Cache hit for: " << source_url << endl);
233  retrieve_and_cache = effective_url->is_expired();
234  BESDEBUG(MODULE, prolog << "Cached target URL is " << (retrieve_and_cache?"":"not ") << "expired." << endl);
235  }
236  // It not found or expired, reload.
237  if(retrieve_and_cache){
238  BESDEBUG(MODULE, prolog << "Acquiring effective URL for " << source_url << endl);
239 
240  {
241  BESStopWatch sw;
242  if(BESDebug::IsSet(MODULE) || BESDebug::IsSet(TIMING_LOG_KEY)) sw.start(prolog + "Retrieve and cache effective url for source url: " + source_url);
243  effective_url = curl::retrieve_effective_url(source_url);
244  }
245  BESDEBUG(MODULE, prolog << " source_url: " << source_url << endl);
246  BESDEBUG(MODULE, prolog << "effective_url: " << effective_url->dump() << endl);
247 
248  d_effective_urls[source_url] = effective_url;
249 
250  BESDEBUG(MODULE, prolog << "Updated record for "<< source_url << " cache size: " << d_effective_urls.size() << endl);
251  }
252  effective_url_str = effective_url->str();
253  BESDEBUG(MODULE_DUMPER, prolog << "dump: " << endl << dump() << endl);
254  } // EucLock dat_lock is released when the point of execution reaches this brace and dat_lock goes out of scope.
255  else {
256  BESDEBUG(MODULE, prolog << "CACHE IS DISABLED." << endl);
257  }
258  BESDEBUG(MODULE, prolog << "END" << endl);
259  return effective_url_str;
260 }

267 bool EffectiveUrlCache::is_enabled()
268 {
269  // The first time here, the value of d_enabled is -1. Once we check for it in TheBESKeys
270  // The value will be 0 (false) or 1 (true) and TheBESKeys will not be checked again.
271  if(d_enabled < 0){
272  bool found;
273  string value;
274  TheBESKeys::TheKeys()->get_value(HTTP_CACHE_EFFECTIVE_URLS_KEY,value,found);
275  BESDEBUG(MODULE, prolog << HTTP_CACHE_EFFECTIVE_URLS_KEY <<": '" << value << "'" << endl);
276  d_enabled = found && BESUtil::lowercase(value)=="true";
277  }
278  BESDEBUG(MODULE, prolog << "d_enabled: " << (d_enabled?"true":"false") << endl);
279  return d_enabled;
280 }
