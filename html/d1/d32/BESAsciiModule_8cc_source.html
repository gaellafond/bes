67 void BESAsciiModule::initialize(const string &modname)
68 {
69  BESDEBUG("ascii", "Initializing module " << modname << endl);
70 
71  BESRequestHandler *handler = new BESAsciiRequestHandler(modname);
72  BESRequestHandlerList::TheList()->add_handler(modname, handler);
73 
74  BESReturnManager::TheManager()->add_transmitter(ASCII_TRANSMITTER, new BESAsciiTransmit());

77  BESDebug::Register("ascii");

79  BESDEBUG("ascii", "Done Initializing module " << modname << endl);
