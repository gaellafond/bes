65  // trigger activation
66  if (check == _SQLH_ON_ALWAYS) {
67  BESDEBUG(SQL_NAME, "SQLErrorManager: This check point is forced, running error checks." << std::endl);
68  /*
69  * run using:
70  * - passed error_factory
71  * - implemented merge result function
72  * - implemented join result function
73  * - do not call getArgs for each action of the list
74  * @see SQLActionManager doc
75  */
76  return SQLActionManager<JOIN, JOIN, JOIN>::doActions(error_factory, &merge, &join, false);
77  }
78  else if (SQLCheckPoint::check(check)) {
79  BESDEBUG(SQL_NAME, "SQLErrorManager: This check point is active, running error checks." << std::endl);
80  /*
81  * run using:
82  * - passed error_factory
83  * - implemented merge result function
84  * - implemented join result function
85  * - do not call getArgs for each action of the list
86  * @see SQLActionManager doc
87  */
88  return SQLActionManager<JOIN, JOIN, JOIN>::doActions(error_factory, &merge, &join, false);
89  }
90  else
91  BESDEBUG(SQL_NAME, "SQLErrorManager: This check point isn't active" << std::endl);
