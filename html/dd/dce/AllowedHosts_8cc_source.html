109 bool AllowedHosts::is_allowed(const std::string &candidate_url)
110 {
111  BESDEBUG(MODULE, prolog << "BEGIN candidate_url: " << candidate_url << endl);
112  bool isAllowed = false;
113  const string file_url("file://");
114  const string http_url("http://");
115  const string https_url("https://");
116 
117  // Special case: This allows any file: URL to pass if the URL starts with the default
118  // catalog's path.
119  if (candidate_url.compare(0, file_url.size(), file_url) == 0 /*equals a file url*/) {
120 
121  // Ensure that the file path starts with the catalog root dir.
122  string file_path = candidate_url.substr(file_url.size());
123  BESDEBUG(MODULE, prolog << "file_path: "<< file_path << endl);
124 
125  BESCatalogList *bcl = BESCatalogList::TheCatalogList();
126  string default_catalog_name = bcl->default_catalog_name();
127  BESDEBUG(MODULE, prolog << "Searching for catalog: "<< default_catalog_name << endl);
128  BESCatalog *bcat = bcl->find_catalog(default_catalog_name);
129  if (bcat) {
130  BESDEBUG(MODULE, prolog << "Found catalog: "<< bcat->get_catalog_name() << endl);
131  }
132  else {
133  string msg = "OUCH! Unable to locate default catalog!";
134  BESDEBUG(MODULE, prolog << msg << endl);
135  throw BESInternalError(msg, __FILE__, __LINE__);
136  }
137 
138  string catalog_root = bcat->get_root();
139  BESDEBUG(MODULE, prolog << "Catalog root: "<< catalog_root << endl);
140 
141  // Never a relative path shall be accepted.
142  // change??
143  // if( file_path[0] != '/'){
144  // file_path.insert(0,"/");
145  //}
146 
147  string relative_path;
148  if(file_path[0] == '/') {
149  if(file_path.length() < catalog_root.length()) {
150  isAllowed = false;
151  }
152  else {
153  int ret = file_path.compare(0, string::npos, catalog_root) == 0;
154  BESDEBUG(MODULE, prolog << "file_path.compare(): " << ret << endl);
155  isAllowed = (ret==0);
156  relative_path = file_path.substr(catalog_root.length());
157  }
158  }
159  else {
160  BESDEBUG(MODULE, prolog << "Relative path detected");
161  relative_path = file_path;
162  isAllowed = true;
163  }
164 
165  // string::compare() returns 0 if the path strings match exactly.
166  // And since we are just looking at the catalog.root as a prefix of the resource
167  // name we only allow access to the resource for an exact match.
168  if(isAllowed) {
169  // If we stop adding a '/' to file_path values that don't begin with one
170  // then we need to detect the use of the relative path here
171  bool follow_sym_links = bcat->get_catalog_utils()->follow_sym_links();
172  try {
173  BESUtil::check_path(relative_path, catalog_root, follow_sym_links);
174  }
175  catch (BESNotFoundError &e) {
176  isAllowed=false;
177  }
178  catch (BESForbiddenError &e) {
179  isAllowed=false;
180  }
181  }
182  BESDEBUG(MODULE, prolog << "File Access Allowed: "<< (isAllowed?"true ":"false ") << endl);
183  }
184  else {
185  // We assume it's an http(s) URL.
186  vector<string>::const_iterator it = d_allowed_hosts.begin();
187  vector<string>::const_iterator end_it = d_allowed_hosts.end();
188  for (; it != end_it && !isAllowed; it++) {
189  string a_regex_pattern = *it;
190  BESRegex reg_expr(a_regex_pattern.c_str());
191  int match_result = reg_expr.match(candidate_url.c_str(), candidate_url.length());
192  if(match_result>=0) {
193  auto match_length = (unsigned int) match_result;
194  if (match_length == candidate_url.length()) {
195  BESDEBUG(MODULE,
196  prolog << "FULL MATCH. pattern: " << a_regex_pattern << " url: " << candidate_url << endl);
197  isAllowed = true;;
198  } else {
199  BESDEBUG(MODULE,
200  prolog << "No Match. pattern: " << a_regex_pattern << " url: " << candidate_url << endl);
201  }
202  }
203  }
204  BESDEBUG(MODULE, prolog << "HTTP Access Allowed: "<< (isAllowed?"true ":"false ") << endl);
205  }
206  BESDEBUG(MODULE, prolog << "END Access Allowed: "<< (isAllowed?"true ":"false ") << endl);
207  return isAllowed;
208 }
