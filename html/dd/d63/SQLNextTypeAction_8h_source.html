157  static libdap::BaseType *
158  nextType(SQLSimpleConnector<SQL_TYPE, ODBC_TYPE> &connector,
159  SQLActionFactory<SQL_TYPE, SQL_TYPE, libdap::BaseType> &type_factory,
160  SQLActionFactory<ERROR_TYPE, ARGS_TYPE, OUT> *error_factory, bool force = false) {
161  try {
162  libdap::BaseType *bt = SQLTypeManager::doActions(type_factory);
163 
169  size_t rows = connector.setNext();
170 
171  // check errors (if active)
172  if (error_factory) {
173  if (force)
174  SQLErrorManager<OUT>::trigger(_SQLH_ON_ALWAYS, *error_factory);
175  else {
176  if (connector.notEnd()) { // check end condition
177  // for each passed row
178  while (rows-- > 0) {
179  // test ON_NEXT_ROW trigger and do actions
180  SQLErrorManager<OUT>::trigger(_SQLH_ON_NEXT_ROW, *error_factory);
181  }
182  }
183  // test ON_GET_NEXT trigger
184  SQLErrorManager<OUT>::trigger(_SQLH_ON_GET_NEXT, *error_factory);
185  }
186  }
187 
188  return bt;
189  }
190  catch (SQLInternalError &ie) {
196  BESDEBUG(SQL_NAME, ie.get_message() << " file:" << ie.get_file() << " row:" << ie.get_line() << endl);
197  BESDEBUG(SQL_NAME, "SQLNextTypeAction: Element skipped" << endl);
198  return NULL;
199  }
200  catch (BESError &e) {
201  connector.close();
202  throw SQLInternalFatalError(e.get_message(), e.get_file(), e.get_line());
203  }
204  catch (exception &e) {
205  connector.close();
206  throw SQLInternalFatalError(e.what(),
207  __FILE__, __LINE__);
208  }
209  catch (...) {
210  connector.close();
211  throw SQLInternalFatalError("SQLNextTypeAction: Unknow error while connecting",
212  __FILE__, __LINE__);
213  }
214  return NULL; // to avoid warning
215  }
