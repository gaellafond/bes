40 SampleSayCommand::parse_request(BESTokenizer &tokenizer, BESDataHandlerInterface &dhi)
41 {

44  /* No sub command, so proceed with the default command

46  dhi.action = SAY_RESPONSE;
47  BESResponseHandler *retResponse = BESResponseHandlerList::TheList()->find_handler( SAY_RESPONSE);
48  if (!retResponse) {
49  string s = (string) "No response handler for command " + SAY_RESPONSE;
50  throw BESSyntaxUserError(s, __FILE__, __LINE__);
51  }
52 
53  my_token = tokenizer.get_next_token();
54  if (my_token == ";") {
55  tokenizer.parse_error(my_token + " not expected\n");
56  }
57 
58  // Here is where your code would parse the tokens
59  dhi.data[SAY_WHAT] = my_token;
60 
61  // Next token should be the token "to"
62  my_token = tokenizer.get_next_token();
63  if (my_token != "to") {
64  tokenizer.parse_error(my_token + " not expected\n");
65  }
66 
67  // Next token should be what is being said
68  my_token = tokenizer.get_next_token();
69  if (my_token == ";") {
70  tokenizer.parse_error(my_token + " not expected\n");
71  }
72  dhi.data[SAY_TO] = my_token;
73 
74  // Last token should be the terminating semicolon (;)
75  my_token = tokenizer.get_next_token();
76  if (my_token != ";") {
77  tokenizer.parse_error(my_token + " not expected\n");
78  }
79 
80  return retResponse;
81 }
