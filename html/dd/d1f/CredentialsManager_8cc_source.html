250 bool file_is_secured(const string &filename) {
251  struct stat st;
252  if (stat(filename.c_str(), &st) != 0) {
253  string err;
254  err.append("file_is_secured() Unable to access file ");
255  err.append(filename).append(" strerror: ").append(strerror(errno));
256  throw BESInternalError(err, __FILE__, __LINE__);
257  }
258 
259  mode_t perm = st.st_mode;
260  bool status;
261  status = (perm & S_IRUSR) && !(
262  // (perm & S_IWUSR) || // We don't need to enforce user no write
263  (perm & S_IXUSR) ||
264  (perm & S_IRGRP) ||
265  (perm & S_IWGRP) ||
266  (perm & S_IXGRP) ||
267  (perm & S_IROTH) ||
268  (perm & S_IWOTH) ||
269  (perm & S_IXOTH));
270  BESDEBUG(CREDS, prolog << "file_is_secured() " << filename << " secured: " << (status ? "true" : "false") << endl);
271  return status;
272 }

304 void CredentialsManager::load_credentials( ) {
305 
306  // This lock is a RAII implementation. It will block until the mutex is
307  // available and the lock will be released when the instance is destroyed.
308  std::lock_guard<std::recursive_mutex> lock_me(d_lock_mutex);
309 
310  bool found_key = true;
311  AccessCredentials *accessCredentials;
312  map<string, AccessCredentials *> credential_sets;
313 
314  string config_file;
315  TheBESKeys::TheKeys()->get_value(CATALOG_MANAGER_CREDENTIALS, config_file, found_key);
316  if(!found_key){
317  BESDEBUG(CREDS, prolog << "The BES key " << CATALOG_MANAGER_CREDENTIALS
318  << " was not found in the BES configuration tree. No AccessCredentials were loaded" << endl);
319  return;
320  }
321 
322  // Does the configuration indicate that credentials will be submitted via the runtime environment?
323  if(config_file == string(CredentialsManager::USE_ENV_CREDS_KEY_VALUE)){
324  // Apparently so...
325  accessCredentials = theCM()->load_credentials_from_env();
326  if(accessCredentials){
327  // So if we have them, we add them to theCM() and then return without processing the configuration.
328  string url = accessCredentials->get(AccessCredentials::URL_KEY);
329  theCM()->add(url,accessCredentials);
330  }
331  // Environment injected credentials override all other configuration credentials.
332  // Since the value of CATALOG_MANAGER_CREDENTIALS is ENV_CREDS_VALUE there is no
333  // Configuration file identified, so wether or not valid credentials information was
334  // found in the ENV we simply return.
335  return;
336  }
337 
338  load_ngap_s3_credentials();
339 
340  if(!file_exists(config_file)){
341  BESDEBUG(CREDS, prolog << "The file specified by the BES key " << CATALOG_MANAGER_CREDENTIALS
342  << " does not exist. No Access Credentials were loaded." << endl);
343  return;
344  }
345 
346  if (!file_is_secured(config_file)) {
347  string err;
348  err.append("CredentialsManager config file ");
349  err.append(config_file);
350  err.append(" is not secured! ");
351  err.append("Set the access permissions to -rw------- (600) and try again.");
352  throw BESInternalError(err, __FILE__, __LINE__);
353  }
354  BESDEBUG(CREDS, prolog << "The config file '" << config_file << "' is secured." << endl);
355 
356  map <string, vector<string>> keystore;
357 
358  kvp::load_keys(config_file, keystore);
359 
360  for(map <string, vector<string>>::iterator it=keystore.begin(); it!=keystore.end(); it++) {
361  string creds_name = it->first;
362  vector<string> &credentials_entries = it->second;
363  map<string, AccessCredentials *>::iterator mit;
364  mit = credential_sets.find(creds_name);
365  if (mit != credential_sets.end()) { // New?
366  // Nope.
367  accessCredentials = mit->second;
368  } else {
369  // Make new one
370  accessCredentials = new AccessCredentials(creds_name);
371  credential_sets.insert(pair<string, AccessCredentials *>(creds_name, accessCredentials));
372  }
373  for (vector<string>::iterator jt = credentials_entries.begin(); jt != credentials_entries.end(); jt++) {
374  string credentials_entry = *jt;
375  int index = credentials_entry.find(":");
376  if (index > 0) {
377  string key_name = credentials_entry.substr(0, index);
378  string value = credentials_entry.substr(index + 1);
379  BESDEBUG(CREDS, prolog << creds_name << ":" << key_name << "=" << value << endl);
380  accessCredentials->add(key_name, value);
381  }
382  }
383  }
384  BESDEBUG(CREDS, prolog << "Loaded " << credential_sets.size() << " AccessCredentials" << endl);
385  vector<AccessCredentials *> bad_creds;
386  map<string,AccessCredentials *>::iterator acit;
387 
388  for (acit = credential_sets.begin(); acit != credential_sets.end(); acit++) {
389  accessCredentials = acit->second;
390  string url = accessCredentials->get(AccessCredentials::URL_KEY);
391  if(url.length()){
392  theCM()->add(url,accessCredentials);
393  }
394  else {
395  bad_creds.push_back(acit->second);
396  }
397  }
398  if(bad_creds.size()){
399  stringstream ss;
400  vector<AccessCredentials * >::iterator bc;
401 
402  ss << "Encountered " << bad_creds.size() << " AccessCredentials "
403  << " definitions missing an associated URL. offenders: ";
404 
405  for (bc = bad_creds.begin(); bc != bad_creds.end(); bc++) {
406  ss << (*bc)->name() << " ";
407  credential_sets.erase((*bc)->name());
408  delete *bc;
409  }
410  throw BESInternalError( ss.str(), __FILE__, __LINE__);
411  }
412  BESDEBUG(CREDS, prolog << "Successfully ingested " << theCM()->size() << " AccessCredentials" << endl);
413 
414 }

425 AccessCredentials *CredentialsManager::load_credentials_from_env( ) {
426 
427  // This lock is a RAII implementation. It will block until the mutex is
428  // available and the lock will be released when the instance is destroyed.
429  std::lock_guard<std::recursive_mutex> lock_me(d_lock_mutex);
430 
431  AccessCredentials *ac = nullptr;
432  string env_url, env_id, env_access_key, env_region, env_bucket;
433 
434  // If we are in developer mode then we compile this section which
435  // allows us to inject credentials via the system environment
436 
437  env_id.assign( get_env_value(CredentialsManager::ENV_ID_KEY));
438  env_access_key.assign(get_env_value(CredentialsManager::ENV_ACCESS_KEY));
439  env_region.assign( get_env_value(CredentialsManager::ENV_REGION_KEY));
440  //env_bucket.assign( get_env_value(CredentialsManager::ENV_BUCKET_KEY));
441  env_url.assign( get_env_value(CredentialsManager::ENV_URL_KEY));
442 
443  if(env_url.length() &&
444  env_id.length() &&
445  env_access_key.length() &&
446  // env_bucket.length() &&
447  env_region.length() ){
448  ac = new AccessCredentials();
449  ac->add(AccessCredentials::URL_KEY, env_url);
450  ac->add(AccessCredentials::ID_KEY, env_id);
451  ac->add(AccessCredentials::KEY_KEY, env_access_key);
452  ac->add(AccessCredentials::REGION_KEY, env_region);
453  // ac->add(AccessCredentials::BUCKET_KEY, env_bucket);
454  }
455  return ac;
456 }
