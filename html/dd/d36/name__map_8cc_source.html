68 string
69 name_map::lookup(string name, const bool canonical_names)
70 {
71  // NEItor is the name_eqiv interator. 7/23/99 jhrg
72  // Could use find_if here. 9/12/2001 jhrg
73  for (NEItor p = _names.begin(); p != _names.end(); ++p)
74  if (name == p->from) {
75  if (!canonical_names) {
76  return p->to;
77  }
78  else {
79  static Regex ident("[A-Za-z_][A-Za-z0-9_]*", 1);
80  string tmp_n = p->to;
81  if (ident.match(tmp_n.c_str(), tmp_n.length()))
82  return munge(tmp_n);
83  else
84  return tmp_n;
85  }
86  }
87 
88  if (!canonical_names)
89  return name;
90  else
91  return munge(name);
92 }
