275  static bool sql_build_dds(SQLDataHandlerInterface &dhi,
276  SQLActionFactory<ERROR_TYPE, MSG_TYPE, OUT_TYPE1> *error_factory,
277  SQLActionFactory<SQL_TYPE, SQL_TYPE, libdap::BaseType> &type_factory,
278  SQLSimpleConnector<SQL_TYPE, ODBC_TYPE> *connector) {
279  BESDEBUG(SQL_NAME, "SQLBuildDDS: Connecting " << endl);
280  SQLConnectAction<ERROR_TYPE, MSG_TYPE, OUT_TYPE1>::connect(
281  // SQLHandleConnector
282  *connector,
283  // get container from storage
284  dhi.getSQLContainer(),
285  // check action to do on errors
286  error_factory); // try to connect
287  BESDEBUG(SQL_NAME, "SQLBuildDDS: Connected" << endl);
288 
289  BESDEBUG(SQL_NAME, "SQLBuildDDS: Executing query" << endl);
290  SQLQueryAction<ERROR_TYPE, MSG_TYPE, OUT_TYPE1>::query(*connector, error_factory);
291  BESDEBUG(SQL_NAME, "SQLBuildDDS: Query successfully done" << endl);
292 
293  BESResponseObject *response = dhi.getResponseObject();
294 
295  BESDDSResponse *bdds = dynamic_cast<BESDDSResponse *>(response);
296  if (!bdds)
297  throw BESInternalFatalError("SQLBuildDDS: Unable to get DDSResponse object", __FILE__, __LINE__);
298 
299  libdap::DDS *dds = bdds->get_dds();
300 
301  // The name of the Sequence was being set to
302  // dhi.getBesContainer()->get_symbolic_name() but I changed that
303  // to the DSN because it seemed like a better name. The symbolic
304  // name was, in fact, the name of the last column of the select
305  // statement, and that seemed like a poor choice. Using the table
306  // name(s) would be awkward and would also 'leak' information about
307  // the database. 9/10/12 jhrg
308  SQLSequence<SQL_TYPE, ODBC_TYPE> *seq
309  = new SQLSequence<SQL_TYPE, ODBC_TYPE>(connector->getParams().getServer(),
310  connector->getParams().get_real_name(), connector);
311 
312  libdap::AttrTable &attr = dds->get_attr_table();
313 
317  connector->reset();
318 
323  for (size_t i = 0; i < connector->getCols(); i++) {
324  libdap::BaseType *bt = NULL;
325 
326  try {
327  BESDEBUG(SQL_NAME, "SQLBuildDDS: getting next object" << endl);
328 
329  bt = SQLNextTypeAction<SQL_TYPE, ODBC_TYPE, ERROR_TYPE, MSG_TYPE, OUT_TYPE1>::nextType(*connector,
330  type_factory,
331  error_factory);
332 
333  if (bt) { // if 'bt' is created
334  BESDEBUG(SQL_NAME, "SQLBuildDDS: done" << endl);
335  // FIXME connector->getColDesc(i) is not the correct type of value
336  // for most of the variables.
337  attr.append_attr(bt->name(), bt->type_name(), connector->getColDesc(i));
338  seq->add_var_nocopy(bt);
339  }
340  else // error occurred on 'bt' creation
341  {
342  BESDEBUG(SQL_NAME, "SQLBuildDDS: : Unrecognized NULL object, Adding default place-holder." << endl);
354  bt = new SQLDummySimpleType(_SQLH_DEFAULT_DAS_NAME, dummy_cast, false);
355  connector->setNext();
356  if (bt) {
357  seq->add_var_nocopy(bt, libdap::nil);
358 
359  // TODO Check on these values. cross check in BuildDAS, too.
360  attr.append_attr(_SQLH_DEFAULT_DAS_NAME, _SQLH_DEFAULT_DAS_TYPE, _SQLH_DEFAULT_DAS_VAL);
361  }
362  else
363  throw BESInternalError("SQLBuildDDS: Unable to allocate SQLObjects", __FILE__, __LINE__);
364  }
365  } // end try
366  catch (BESError &e) {
367  BESDEBUG(SQL_NAME, "SQLBuildDDS: Caught error: " << e.get_message() << endl);
368  // FIXME BESLog
369  //cerr << "SQLBuildDDS: Caught error: " << e.get_message() << endl;
370  throw;
371  }
372  catch (...) {
373  BESDEBUG(SQL_NAME, "SQLBuildDDS: Caught something" << endl);
374  // FIXME BESLog
375  //cerr << "caught something" << endl;
376  throw;
377  }
378  } // endfor col
379 
380  // reset connector cursor position before start transfer
381  connector->reset(); // done by SQLSequence constructor
382 
383  if (seq) {
384  BESDEBUG(SQL_NAME, "SQLBuildDDS: Adding variable to dds" << endl);
385  dds->add_var_nocopy(seq);
386 
387  // The BES is setting this to 'virtual'. This is a better
388  // name because it is consistent with how other responses
389  // are named. 9/10/12 jhrg
390  string name = connector->getParams().get_real_name();
391  size_t p = name.find_last_of('/');
392  if (p != string::npos && p + 1 <= name.length())
393  name = name.substr(p + 1);
394  dds->set_dataset_name(name);
395  }
396  else
397  throw BESInternalError("Unable to build an SQLSequence", __FILE__, __LINE__);
398 
399  dhi.getBesHandler().data[POST_CONSTRAINT] = dhi.getBesContainer()->get_constraint();
400 
401  BESDEBUG(SQL_NAME, "SQLBuildDDS: Set constraint" << endl);
402  bdds->set_constraint(dhi.getBesHandler());
403 
404  BESDEBUG(SQL_NAME, "SQLBuildDDS: Clear container" << endl);
405  bdds->clear_container();
406 
407  BESDEBUG(SQL_NAME, "SQLBuildDDS: DDS is built" << endl);
408  return true;
409  }
