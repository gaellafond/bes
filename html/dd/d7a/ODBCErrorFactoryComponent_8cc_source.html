65 SQLActionList<message_t, void> &
66 ODBCErrorFactoryComponent::_getActions(error_t *error) {
67  BESDEBUG(ODBC_NAME, "Checking for new errors.." << endl);
68  if (!error)
69  throw BESInternalFatalError(
70  "ODBCPlugin: error code is NULL", __FILE__, __LINE__);
71  switch (*error) {
72  case SQL_SUCCESS:
73  BESDEBUG(ODBC_NAME, "SUCCESS" << endl);
74  return ODBCErrorFactoryComponent::success;
75  break;
76  case SQL_SUCCESS_WITH_INFO:
77  BESDEBUG(ODBC_NAME, "INFO" << endl);
78  return ODBCErrorFactoryComponent::info;
79  break;
80  case SQL_NO_DATA:
81  BESDEBUG(ODBC_NAME, "SQL_NO_DATA" << endl);
82  return ODBCErrorFactoryComponent::no_data;
83  break;
84  case SQL_ERROR:
85  BESDEBUG(ODBC_NAME, "SQL_ERROR" << endl);
86  return ODBCErrorFactoryComponent::error;
87  break;
88  case SQL_INVALID_HANDLE:
89  BESDEBUG(ODBC_NAME, "SQL_INVALID_HANDLE" << endl);
90  return ODBCErrorFactoryComponent::invalid;
91  break;
92  default:
93  BESDEBUG(ODBC_NAME, "NOT MAPPED ERROR" << endl);
94  throw BESInternalFatalError("No ErrorList found", __FILE__, __LINE__);
95  }
96 }
