81 void FONcStr::define(int ncid)
82 {
83  if (!_defined) {
84  BESDEBUG("fonc", "FONcStr::define - defining " << _varname << endl);
85 
86  _varname = FONcUtils::gen_name(_embed, _varname, _orig_varname);
87  _data = new string;
88  _str->buf2val((void**) &_data);
89  int size = _data->size() + 1;
90 
91  string dimname;
92 
93  // For DAP4, we need to ensure the dimension name is unique.
94  if(is_dap4_group == true) {
95  ostringstream dim_suffix_strm;
96  dim_suffix_strm <<"_len"<<FONcDim::DimNameNum +1;
97  FONcDim::DimNameNum++;
98  dimname = _varname+dim_suffix_strm.str();
99  }
100  else
101  dimname = _varname + "_len";
102  int stax = nc_def_dim(ncid, dimname.c_str(), size, &_dimid);
103  if (stax != NC_NOERR) {
104  string err = (string) "fileout.netcdf - " + "Failed to define dim " + dimname + " for " + _varname;
105  FONcUtils::handle_error(stax, err, __FILE__, __LINE__);
106  }
107 
108  int var_dims[1]; // variable shape
109  var_dims[0] = _dimid;
110  stax = nc_def_var(ncid, _varname.c_str(), NC_CHAR, 1, var_dims, &_varid);
111  if (stax != NC_NOERR) {
112  string err = (string) "fileout.netcdf - " + "Failed to define var " + _varname;
113  FONcUtils::handle_error(stax, err, __FILE__, __LINE__);
114  }
115 
116  _defined = true;
117 
118  FONcAttributes::add_variable_attributes(ncid, _varid, _str,isNetCDF4_ENHANCED(),is_dap4);
119  FONcAttributes::add_original_name(ncid, _varid, _varname, _orig_varname);
120 
121  BESDEBUG("fonc", "FONcStr::define - done defining " << _varname << endl);
122  }
123 }

134 void FONcStr::write(int ncid)
135 {
136  BESDEBUG("fonc", "FONcStr::write for var " << _varname << endl);
137 
138  size_t var_start[1]; // variable start
139  size_t var_count[1]; // variable count
140 
141  var_count[0] = _data->size() + 1;
142  var_start[0] = 0;
143  int stax = nc_put_vara_text(ncid, _varid, var_start, var_count, _data->c_str());
144  if (stax != NC_NOERR) {
145  string err = (string) "fileout.netcdf - " + "Failed to write string data " + *_data + " for " + _varname;
146  delete _data;
147  _data = 0;
148  FONcUtils::handle_error(stax, err, __FILE__, __LINE__);
149  }
150  delete _data;
151  _data = 0;
152 
153  BESDEBUG("fonc", "FONcStr::done write for var " << _varname << endl);
154 }
