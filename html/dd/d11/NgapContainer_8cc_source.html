157  string NgapContainer::access() {
158  BESDEBUG(MODULE, prolog << "BEGIN object address: "<< (void *) this << endl);
159 
160  // Since this the ngap we know that the real_name is a URL.
161  string data_access_url_str = get_real_name();
162 
163  // And we know that the dmr++ file should "right next to it" (side-car)
164  string dmrpp_url = data_access_url_str + ".dmrpp";
165 
166  // And if there's a missing data file (side-car) it should be "right there" too.
167  string missing_data_url = data_access_url_str + ".missing";
168 
169  BESDEBUG(MODULE, prolog << " data_access_url: " << data_access_url_str << endl);
170  BESDEBUG(MODULE, prolog << " dmrpp_url: " << dmrpp_url << endl);
171  BESDEBUG(MODULE, prolog << "missing_data_url: " << missing_data_url << endl);
172 
173  string type = get_container_type();
174  if (type == "ngap")
175  type = "";
176 
177  if (!d_dmrpp_rresource) {
178  BESDEBUG(MODULE, prolog << "Building new RemoteResource (dmr++)." << endl);
179  map<string,string> content_filters;
180  if (inject_data_url()) {
181  content_filters.insert(pair<string,string>(DATA_ACCESS_URL_KEY,data_access_url_str));
182  content_filters.insert(pair<string,string>(MISSING_DATA_ACCESS_URL_KEY,missing_data_url));
183  }
184  {
185  d_dmrpp_rresource = new http::RemoteResource(dmrpp_url);
186  BESStopWatch besTimer;
187  if (BESISDEBUG(MODULE) || BESDebug::IsSet(TIMING_LOG_KEY) || BESLog::TheLog()->is_verbose()){
188  besTimer.start("DMR++ retrieval: "+ dmrpp_url);
189  }
190  d_dmrpp_rresource->retrieveResource(content_filters);
191  }
192  }
193  BESDEBUG(MODULE, prolog << "Retrieved remote resource: " << dmrpp_url << endl);
194 
195  // TODO This file should be read locked before leaving this method.
196  string cachedResource = d_dmrpp_rresource->getCacheFileName();
197  BESDEBUG(MODULE, prolog << "Using local cache file: " << cachedResource << endl);
198 
199  type = d_dmrpp_rresource->getType();
200  set_container_type(type);
201  BESDEBUG(MODULE, prolog << "Type: " << type << endl);
202  BESDEBUG(MODULE, prolog << "Done retrieving: " << dmrpp_url << " returning cached file " << cachedResource << endl);
203  BESDEBUG(MODULE, prolog << "END" << endl);
204 
205  return cachedResource; // this should return the dmr++ file name from the NgapCache
206  }
