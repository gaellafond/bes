121 bool WWWStructure::is_simple_structure()
122 {
123  for (Vars_iter i = var_begin(); i != var_end(); ++i) {
124  if ((*i)->type() == dods_structure_c) {
125  if (!dynamic_cast<WWWStructure *> (*i)->is_simple_structure())
126  return false;
127  }
128  else {
129  if (!(*i)->is_simple_type())
130  return false;
131  }
132  }
133 
134  return true;
