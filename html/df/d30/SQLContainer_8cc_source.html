60 bool
61 SQLContainer::isUpToDate() const {
62  struct stat buf;
63  if (stat(this->get_real_name().c_str(), &buf) == 0) {
64  if (fabs(difftime(_time, buf.st_mtime)) < _SQLH_CONT_TIME_DIF) {
65  BESDEBUG(SQL_NAME, "SQLContainer: Cached file is up to date" << endl);
66  return true;
67  }
68  else {
69  BESDEBUG(SQL_NAME,
70  "SQLContainer: Cached file is old and should be updated" << endl);
71  return false;
72  }
73  }
74  else {
75  BESDEBUG(SQL_NAME,
76  "SQLContainer: Unable to get time/date modification" << endl);
77  return false;
78  }
79 }
