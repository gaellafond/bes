51 bool SQLCheckPoint::init() throw(BESError) {
52  if (!isSet) {
53  last_status = false; // false by default
54  last_type_checked = _SQLH_CHECK_SIZE + 1; // never checked
55  vector<string> _set_check_p;
56  const string &key = string(_SQLH_CHECK_KEY);
57  bool found = false;
58  TheBESKeys::TheKeys()->get_values(key, _set_check_p, found);
59  if (!found)
60  throw BESInternalError("Unable to find \"" + key + "\" key into configuration file", __FILE__, __LINE__);
61 
62  for (std::vector<string>::iterator i = _set_check_p.begin(); i != _set_check_p.end(); i++) {
63  list<std::string> tocheck;
64  BESUtil::explode(_SQLH_CHECK_SEPARATOR, i->c_str(), tocheck);
65 
66  for (list<string>::iterator j = tocheck.begin(); j != tocheck.end(); j++) {
67  int val = atoi((j->c_str()));
68  if (val < _SQLH_CHECK_SIZE && val > 0)
69  function.set(val, true);
70  else
71  throw BESInternalError("Bad value into \"" + string(_SQLH_CHECK_KEY) + "\" key set.", __FILE__,
72  __LINE__);
73  }
74  }
75  isSet = true;
76  } //if !isSet
77 
78  // The docs say this always returns true; I think it could 'return' void
79  // but 'return true;' might be a better match for likely uses. jhrg 10/1/14
80  return true;
81 }
