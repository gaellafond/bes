278 bool SQLRequestHandler::update_wrap_count(const string &name, bool add) throw(BESInternalFatalError) {
279  std::map<string, size_t>::iterator wi = theWrapCount().find(name);
280  if (wi != theWrapCount().end()) {
281  if (add) {
282  ++((*wi).second);
283  BESDEBUG(SQL_NAME, "SQLRequestHandler: wrapper " << name << " use_count: " << (*wi).second << endl);
284  }
285  else { //add==false -> remove
286  if ((--((*wi).second)) == 0) {
287  BESDEBUG(SQL_NAME,
288  "SQLRequestHandler: wrapper " << name << " use_count: 1 and is going to be erased" << endl);
289  theWrapCount().erase(wi);
290  return true;
291  }
292  }
293  }
294  else {
295  if (add) {
296  BESDEBUG(SQL_NAME, "SQLRequestHandler: wrapper " << name << " use_count: 1" << endl);
297  theWrapCount().insert(std::make_pair(name, 1));
298  }
299  else
300  throw BESInternalFatalError("This handler seams to be not correctly wrapped!",
301  __FILE__, __LINE__);
302  }
303  return false;
304 }
