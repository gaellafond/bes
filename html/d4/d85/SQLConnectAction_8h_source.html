150  static OUT *
151  connect(SQLHandleConnector &connector,
152  SQLContainer *container,
153  SQLActionFactory<ERROR_TYPE, MSG_TYPE, OUT> *error_factory,
154  bool force = false) {
155  BESDEBUG(SQL_NAME, "SQLConnectAction: Connecting" << std::endl);
156  try {
157  bool ret = connector.connect(container);
158  if (error_factory) {
159  if (force)
160  return SQLErrorManager<OUT>::trigger(_SQLH_ON_ALWAYS,
161  *error_factory);
162  else
163  return SQLErrorManager<OUT>::trigger(_SQLH_ON_CONNECT,
164  *error_factory);
165  }
166  else if (!ret)
167  throw SQLInternalFatalError(
168  "SQLConnectAction: Connect execution fails (returning false)",
169  __FILE__, __LINE__);
170 
171  BESDEBUG(SQL_NAME, "SQLConnectAction: Connected" << std::endl);
172  }
173  catch (SQLInternalError &ie) {
174  // don't forget to close the connector
175  connector.close();
180  throw SQLInternalFatalError(ie.get_message(),
181  ie.get_file(), ie.get_line());
182  }
183  catch (BESError &e) {
184  // don't forget to close the connector
185  connector.close();
186  throw SQLInternalFatalError(e.get_message(),
187  e.get_file(), e.get_line());
188  }
189  catch (std::exception &e) {
190  // don't forget to close the connector
191  connector.close();
192  throw SQLInternalFatalError(e.what(),
193  __FILE__, __LINE__);
194  }
195  catch (...) {
196  // don't forget to close the connector
197  connector.close();
198  throw SQLInternalFatalError(
199  "SQLConnectAction: Unknown error while connecting",
200  __FILE__, __LINE__);
201  }
202  return NULL; // to avoid warning
203  }
204
