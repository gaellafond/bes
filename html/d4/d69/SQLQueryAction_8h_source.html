146  static OUT *
147  query(SQLHandleConnector &connector,
148  SQLActionFactory<ERROR_TYPE, ARGS_TYPE, OUT> *error_factory,
149  bool force = false) {
150  // try to make a query
151  BESDEBUG(SQL_NAME, "SQLQueryAction: Executing query" << std::endl);
152  try {
156  bool ret = connector.query();
157  if (error_factory) {
158  if (force)
159  SQLErrorManager<OUT>::trigger(_SQLH_ON_ALWAYS,
160  *error_factory);
161  else
162  SQLErrorManager<OUT>::trigger(_SQLH_ON_QUERY,
163  *error_factory);
164  }
165  else if (!ret)
166  throw SQLInternalFatalError(
167  "SQLQueryAction: Query execution fails (returning false)",
168  __FILE__, __LINE__);
169  BESDEBUG(SQL_NAME, "SQLQueryAction: Query successfully done" << std::endl);
170  }
171  catch (SQLInternalError &ie) {
172  connector.close();
173  throw;
174  }
175  catch (SQLInternalFatalError &ie) {
176  connector.close();
177  throw SQLInternalFatalError(ie.get_message(), ie.get_file(),
178  ie.get_line());
179  }
180  catch (std::exception &e) {
181  connector.close();
185  throw SQLInternalFatalError(e.what(), __FILE__, __LINE__);
186  }
187  catch (...) {
188  throw SQLInternalFatalError(
189  "SQLQueryAction: Unknow error while executing query",
190  __FILE__, __LINE__);
191  }
192  return NULL; // to avoid warning
193  }
