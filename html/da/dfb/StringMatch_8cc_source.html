36 StringMatch::match(const char *regex, size_t n_groups, std::vector<size_t> &groups, string &row) {
37  if (!regex) {
38  throw BESInternalFatalError(
39  "StringMatch::match Unable to use NULL regex pointer",
40  __FILE__, __LINE__);
41  }
42  // regex
43  regex_t _reg_expr;
44  // compile regex
45  if (regcomp(&_reg_expr, regex, REG_EXTENDED) != 0) {
46  throw BESInternalError(
47  "StringMatch: Unable to compile the REGEX: " + (string) regex,
48  __FILE__, __LINE__);
49  }
50  // returned vector
51  std::vector<std::vector<string> > res;
55  regmatch_t indexes[n_groups]; // index matches
56  // while regex match into the string
57  while (regexec(&_reg_expr, row.c_str(), n_groups, indexes, 0) == 0) {
58  size_t gr = 0;
59  std::vector<size_t>::iterator it = groups.begin();
60  std::vector<string> group_vect =
61  std::vector<string>(groups.size());
62  // while groups in the list
63  while (it != groups.end()) {
64  // check if limits are acceptable (group is found)
65  if (indexes[*it].rm_so != -1) // if some group is found
66  group_vect[gr] =
67  row.substr(
68  indexes[*it].rm_so,
69  indexes[*it].rm_eo - indexes[*it].rm_so);
70 
71  // next group to match
72  it++;
73  // store in position
74  gr++;
75  }
76  // delete matching string (entire string)
77  row.erase(indexes[0].rm_so, indexes[0].rm_eo - indexes[0].rm_so);
78  // store found groups
79  res.push_back(group_vect);
80  }
81  regfree(&_reg_expr);
82 
83  return res;
84 }
