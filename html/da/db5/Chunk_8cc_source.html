70  // received header is nitems * size long in 'buffer' NOT ZERO TERMINATED
71  // 'userdata' is set with CURLOPT_HEADERDATA

101 size_t chunk_write_data(void *buffer, size_t size, size_t nmemb, void *data) {
102  size_t nbytes = size * nmemb;
103  auto chunk = reinterpret_cast<Chunk *>(data);
104 
105  BESDEBUG(MODULE, prolog << "BEGIN chunk->get_response_content_type():" << chunk->get_response_content_type()
106  << " chunk->get_data_url(): " << chunk->get_data_url() << endl);
107 
108  // When Content-Type is 'application/xml,' that's an error. jhrg 6/9/20
109  if (chunk->get_response_content_type().find("application/xml") != string::npos) {
110  // At this point we no longer care about great performance - error msg readability
111  // is more important. jhrg 12/30/19
112  string xml_message = reinterpret_cast<const char *>(buffer);
113  xml_message.erase(xml_message.find_last_not_of("\t\n\v\f\r 0") + 1);
114  // Decode the AWS XML error message. In some cases this will fail because pub keys,
115  // which maybe in this error text, may have < or > chars in them. the XML parser
116  // will be sad if that happens. jhrg 12/30/19
117  try {
118  string json_message = xml2json(xml_message.c_str());
119  rapidjson::Document d;
120  d.Parse(json_message.c_str());
121  rapidjson::Value &message = d["Error"]["Message"];
122  rapidjson::Value &code = d["Error"]["Code"];
123 
124  // We might want to get the "Code" from the "Error" if these text messages
125  // are not good enough. But the "Code" is not really suitable for normal humans...
126  // jhrg 12/31/19
127 
128  if (string(code.GetString()) == "AccessDenied") {
129  stringstream msg;
130  msg << prolog << "ACCESS DENIED - The underlying object store has refused access to: ";
131  msg << chunk->get_data_url() << " Object Store Message: " << json_message;
132  BESDEBUG(MODULE, msg.str() << endl);
133  VERBOSE(msg.str() << endl);
134  throw BESForbiddenError(msg.str(), __FILE__, __LINE__);
135  }
136  else {
137  stringstream msg;
138  msg << prolog << "ERROR - The underlying object store returned an error. ";
139  msg << "(Tried: " << chunk->get_data_url() << ") Object Store Message: " << json_message;
140  BESDEBUG(MODULE, msg.str() << endl);
141  VERBOSE(msg.str() << endl);
142  throw BESInternalError(msg.str(), __FILE__, __LINE__);
143  }
144  }
145  catch (BESError) {
146  // re-throw any BESError - added for the future if we make BESError a child
147  // of std::exception as it should be. jhrg 12/30/19
148  throw;
149  }
150  catch (std::exception &e) {
151  stringstream msg;
152  msg << prolog << "Caught std::exception when accessing object store data. (Tried: " << chunk->get_data_url() << ")" <<
153  " Message: " << e.what();
154  BESDEBUG(MODULE, msg.str() << endl);
155  throw BESSyntaxUserError(msg.str(), __FILE__, __LINE__);
156  }
157  }
158 
159  // rbuf: |******++++++++++----------------------|
160  // ^ ^ bytes_read + nbytes
161  // | bytes_read
162 
163  unsigned long long bytes_read = chunk->get_bytes_read();
164 
165  // If this fails, the code will write beyond the buffer.
166  if (bytes_read + nbytes > chunk->get_rbuf_size()) {
167  stringstream msg;
168  msg << prolog << "ERROR! The number of bytes_read: " << bytes_read << " plus the number of bytes to read: "
169  << nbytes << " is larger than the target buffer size: " << chunk->get_rbuf_size();
170  BESDEBUG(MODULE, msg.str() << endl);
171  DmrppRequestHandler::curl_handle_pool->release_all_handles();
172  throw BESInternalError(msg.str(), __FILE__, __LINE__);
173  }
174 
175  memcpy(chunk->get_rbuf() + bytes_read, buffer, nbytes);
176  chunk->set_bytes_read(bytes_read + nbytes);
177 
178  BESDEBUG(MODULE, prolog << "END" << endl);
179 
180  return nbytes;
181 }

193 void inflate(char *dest, unsigned long long dest_len, char *src, unsigned long long src_len) {
194  /* Sanity check */
195  assert(src_len > 0);
196  assert(src);
197  assert(dest_len > 0);
198  assert(dest);
199 
200  /* Input; uncompress */
201  z_stream z_strm; /* zlib parameters */
202 
203  /* Set the uncompression parameters */
204  memset(&z_strm, 0, sizeof(z_strm));
205  z_strm.next_in = (Bytef *) src;
206  z_strm.avail_in = src_len;
207  z_strm.next_out = (Bytef *) dest;
208  z_strm.avail_out = dest_len;
209 
210  /* Initialize the uncompression routines */
211  if (Z_OK != inflateInit(&z_strm))
212  throw BESError("Failed to initialize inflate software.", BES_INTERNAL_ERROR, __FILE__, __LINE__);
213 
214  /* Loop to uncompress the buffer */
215  int status = Z_OK;
216  do {
217  /* Uncompress some data */
218  status = inflate(&z_strm, Z_SYNC_FLUSH);
219 
220  /* Check if we are done uncompressing data */
221  if (Z_STREAM_END == status) break; /*done*/
222 
223  /* Check for error */
224  if (Z_OK != status) {
225  (void) inflateEnd(&z_strm);
226  throw BESError("Failed to inflate data chunk.", BES_INTERNAL_ERROR, __FILE__, __LINE__);
227  }
228  else {
229  /* If we're not done and just ran out of buffer space, it's an error.
230  * The HDF5 library code would extend the buffer as needed, but for
231  * this handler, we always know the size of the uncompressed chunk.
232  */
233  if (0 == z_strm.avail_out) {
234  throw BESError("Data buffer is not big enough for uncompressed data.", BES_INTERNAL_ERROR, __FILE__,
235  __LINE__);
236 #if 0
237  /* Here's how to extend the buffer if needed. This might be useful some day... */
238  void *new_outbuf; /* Pointer to new output buffer */
239 
240  /* Allocate a buffer twice as big */
241  nalloc *= 2;
242  if (NULL == (new_outbuf = H5MM_realloc(outbuf, nalloc))) {
243  (void) inflateEnd(&z_strm);
244  HGOTO_ERROR(H5E_RESOURCE, H5E_NOSPACE, 0, "memory allocation failed for inflate decompression")
245  } /* end if */
246  outbuf = new_outbuf;
247 
248  /* Update pointers to buffer for next set of uncompressed data */
249  z_strm.next_out = (unsigned char*) outbuf + z_strm.total_out;
250  z_strm.avail_out = (uInt) (nalloc - z_strm.total_out);
251 #endif
252  } /* end if */
253  } /* end else */
254  } while (status == Z_OK);
255 
256  /* Finish uncompressing the stream */
257  (void) inflateEnd(&z_strm);
258

287 void unshuffle(char *dest, const char *src, unsigned long long src_size, unsigned long long width) {
288  unsigned long long elems = src_size / width; // int division rounds down
289 
290  /* Don't do anything for 1-byte elements, or "fractional" elements */
291  if (!(width > 1 && elems > 1)) {
292  memcpy(dest, const_cast<char *>(src), src_size);
293  }
294  else {
295  /* Get the pointer to the source buffer (Alias for source buffer) */
296  char *_src = const_cast<char *>(src);
297  char *_dest = 0; // Alias for destination buffer
298 
299  /* Input; unshuffle */
300  for (unsigned int i = 0; i < width; i++) {
301  _dest = dest + i;
302 #ifndef DUFFS_DEVICE
303  size_t j = elems;
304  while(j > 0) {
305  *_dest = *_src++;
306  _dest += width;
307 
308  j--;
309  }
310 #else /* DUFFS_DEVICE */
311  {
312  size_t duffs_index = (elems + 7) / 8; /* Counting index for Duff's device */
313  switch (elems % 8) {
314  default:
315  assert(0 && "This Should never be executed!");
316  break;
317  case 0:
318  do {
319  // This macro saves repeating the same line 8 times
320 #define DUFF_GUTS *_dest = *_src++; _dest += width;
321 
322  DUFF_GUTS
323  case 7:
324  DUFF_GUTS
325  case 6:
326  DUFF_GUTS
327  case 5:
328  DUFF_GUTS
329  case 4:
330  DUFF_GUTS
331  case 3:
332  DUFF_GUTS
333  case 2:
334  DUFF_GUTS
335  case 1:
336  DUFF_GUTS
337  } while (--duffs_index > 0);
338  } /* end switch */
339  } /* end block */
340 #endif /* DUFFS_DEVICE */
341 
342  } /* end for i = 0 to width*/
343 
344  /* Compute the leftover bytes if there are any */
345  size_t leftover = src_size % width;
346 
347  /* Add leftover to the end of data */
348  if (leftover > 0) {
349  /* Adjust back to end of shuffled bytes */
350  _dest -= (width - 1); /*lint !e794 _dest is initialized */
351  memcpy((void *) _dest, (void *) _src, leftover);
352  }
353  } /* end if width and elems both > 1 */
354 }

357 void Chunk::parse_chunk_position_in_array_string(const string &pia, vector<unsigned long long> &cpia_vect){
358  if (pia.empty()) return;
359 
360  if (!cpia_vect.empty()) cpia_vect.clear();
361 
362  // Assume input is [x,y,...,z] where x, ..., are integers; modest syntax checking
363  // [1] is a minimal 'position in array' string.
364  if (pia.find('[') == string::npos || pia.find(']') == string::npos || pia.length() < 3)
365  throw BESInternalError("while parsing a DMR++, chunk position string malformed", __FILE__, __LINE__);
366 
367  if (pia.find_first_not_of("[]1234567890,") != string::npos)
368  throw BESInternalError("while parsing a DMR++, chunk position string illegal character(s)", __FILE__, __LINE__);
369 
370  // strip off []; iss holds x,y,...,z
371  istringstream iss(pia.substr(1, pia.length() - 2));
372 
373  char c;
374  unsigned int i;
375  while (!iss.eof()) {
376  iss >> i; // read an integer
377  cpia_vect.push_back(i);
378  iss >> c; // read a separator (,)
379  }
380 }

531 void Chunk::inflate_chunk(bool deflate, bool shuffle, unsigned long long chunk_size, unsigned long long elem_width) {
532  // This code is pretty naive - there are apparently a number of
533  // different ways HDF5 can compress data, and it does also use a scheme
534  // where several algorithms can be applied in sequence. For now, get
535  // simple zlib deflate working.jhrg 1/15/17
536  // Added support for shuffle. Assuming unshuffle always is applied _after_
537  // inflating the data (reversing the shuffle --> deflate process). It is
538  // possible that data could just be deflated or shuffled (because we
539  // have test data are use only shuffle). jhrg 1/20/17
540  // The file that implements the deflate filter is H5Zdeflate.c in the hdf5 source.
541  // The file that implements the shuffle filter is H5Zshuffle.c.
542 
543  if (d_is_inflated)
544  return;
545 
546  chunk_size *= elem_width;
547 
548  if (deflate) {
549  char *dest = new char[chunk_size];
550  try {
551  inflate(dest, chunk_size, get_rbuf(), get_rbuf_size());
552  // This replaces (and deletes) the original read_buffer with dest.
553 #if DMRPP_USE_SUPER_CHUNKS
554  set_read_buffer(dest, chunk_size, chunk_size, true);
555 #else
556  set_rbuf(dest, chunk_size);
557 #endif
558  }
559  catch (...) {
560  delete[] dest;
561  throw;
562  }
563  }
564 
565  if (shuffle) {
566  // The internal buffer is chunk's full size at this point.
567  char *dest = new char[get_rbuf_size()];
568  try {
569  unshuffle(dest, get_rbuf(), get_rbuf_size(), elem_width);
570 #if DMRPP_USE_SUPER_CHUNKS
571  set_read_buffer(dest,get_rbuf_size(),get_rbuf_size(), true);
572 #else
573  set_rbuf(dest, get_rbuf_size());
574 #endif
575  }
576  catch (...) {
577  delete[] dest;
578  throw;
579  }
580  }
581 
582  d_is_inflated = true;
583 
584 #if 0 // This was handy during development for debugging. Keep it for awhile (year or two) before we drop it ndp - 01/18/17
585  if(BESDebug::IsSet(MODULE)) {
586  unsigned long long chunk_buf_size = get_rbuf_size();
