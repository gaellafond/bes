96 bool get_next_future(list<std::future<bool>> &futures, atomic_uint &thread_counter, unsigned long timeout, string debug_prefix) {
97  bool future_finished = false;
98  bool done = false;
99  std::chrono::milliseconds timeout_ms (timeout);
100 
101  while(!done){
102  auto futr = futures.begin();
103  auto fend = futures.end();
104  bool future_is_valid = true;
105  while(!future_finished && future_is_valid && futr != fend){
106  future_is_valid = (*futr).valid();
107  if(future_is_valid){
108  // What happens if wait_for() always returns future_status::timeout for a stuck thread?
109  // If that were to happen, the loop would run forever. However, we assume that these
110  // threads are never 'stuck.' We assume that their computations always complete, either
111  // with success or failure. For the transfer threads, timeouts will stop them if nothing
112  // else does and for the decompression threads, the worst case is a segmentation fault.
113  // jhrg 2/5/21
114  if((*futr).wait_for(timeout_ms) != std::future_status::timeout){
115  try {
116  bool success = (*futr).get();
117  future_finished = true;
118  BESDEBUG(dmrpp_3, debug_prefix << prolog << "Called future::get() on a ready future."
119  << " success: " << (success?"true":"false") << endl);
120  if(!success){
121  stringstream msg;
122  msg << debug_prefix << prolog << "The std::future has failed!";
123  msg << " thread_counter: " << thread_counter;
124  throw BESInternalError(msg.str(), __FILE__, __LINE__);
125  }
126  }
127  catch(...){
128  // TODO I had to add this to make the thread counting work when there's errors
129  // But I think it's primitive because it trashes everything - there's
130  // surely a way to handle the situation on a per thread basis and maybe even
131  // retry?
132  futures.clear();
133  thread_counter=0;
134  throw;
135  }
136  }
137  else {
138  futr++;
139  BESDEBUG(dmrpp_3, debug_prefix << prolog << "future::wait_for() timed out. (timeout: " <<
140  timeout << " ms) There are currently " << futures.size() << " futures in process."
141  << " thread_counter: " << thread_counter << endl);
142  }
143  }
144  else {
145  BESDEBUG(dmrpp_3, debug_prefix << prolog << "The future was not valid. Dumping... " << endl);
146  future_finished = true;
147  }
148  }
149  if (futr!=fend && future_finished) {
150  futures.erase(futr);
151  thread_counter--;
152  BESDEBUG(dmrpp_3, debug_prefix << prolog << "Erased future from futures list. (Erased future was "
153  << (future_is_valid?"":"not ") << "valid at start.) There are currently " <<
154  futures.size() << " futures in process. thread_counter: " << thread_counter << endl);
155  }
156  done = future_finished || futures.empty();
157  }
158  return future_finished;
159 }

173 bool one_child_chunk_thread_new(unique_ptr<one_child_chunk_args_new> args)
174 {
175 
176  args->child_chunk->read_chunk();
177 
178  assert(args->the_one_chunk->get_rbuf());
179  assert(args->child_chunk->get_rbuf());
180  assert(args->child_chunk->get_bytes_read() == args->child_chunk->get_size());
181 
182  // the_one_chunk offset \/
183  // the_one_chunk: mmmmmmmmmmmmmmmm
184  // child chunks: 1111222233334444 (there are four child chunks)
185  // child offsets: ^ ^ ^ ^
186  // For this example, child_1_offset - the_one_chunk_offset == 0 (that's always true)
187  // child_2_offset - the_one_chunk_offset == 4; child_2_offset - the_one_chunk_offset == 8
188  // and child_3_offset - the_one_chunk_offset == 12.

190  // where that child chunk should be written.

194

196  args->child_chunk->get_bytes_read());
197

319 void read_super_chunks_unconstrained_concurrent(queue<shared_ptr<SuperChunk>> &super_chunks, DmrppArray *array)
320 {
321  BESStopWatch sw;
322  if (BESDebug::IsSet(TIMING_LOG_KEY)) sw.start(prolog + " name: "+array->name(), "");
323 
324  // Parallel version based on read_chunks_unconstrained(). There is
325  // substantial duplication of the code in read_chunks_unconstrained(), but
326  // wait to remove that when we move to C++11 which has threads integrated.
327 
328  // We maintain a list of futures to track our parallel activities.
329  list<future<bool>> futures;
330  try {
331  bool done = false;
332  bool future_finished = true;
333  while (!done) {
334 
335  if(!futures.empty())
336  future_finished = get_next_future(futures, transfer_thread_counter, DMRPP_WAIT_FOR_FUTURE_MS, prolog);
337 
338  // If future_finished is true this means that the chunk_processing_thread_counter has been decremented,
339  // because future::get() was called or a call to future::valid() returned false.
340  BESDEBUG(dmrpp_3, prolog << "future_finished: " << (future_finished ? "true" : "false") << endl);
341 
342  if (!super_chunks.empty()){
343  // Next we try to add a new Chunk compute thread if we can - there might be room.
344  bool thread_started = true;
345  while(thread_started && !super_chunks.empty()) {
346  auto super_chunk = super_chunks.front();
347  BESDEBUG(dmrpp_3, prolog << "Starting thread for " << super_chunk->to_string(false) << endl);
348 
349  auto args = unique_ptr<one_super_chunk_args>(new one_super_chunk_args(super_chunk, array));
350  thread_started = start_super_chunk_unconstrained_transfer_thread(futures, std::move(args));
351 
352  if (thread_started) {
353  super_chunks.pop();
354  BESDEBUG(dmrpp_3, prolog << "STARTED thread for " << super_chunk->to_string(false) << endl);
355  } else {
356  // Thread did not start, ownership of the arguments was not passed to the thread.
357  BESDEBUG(dmrpp_3, prolog << "Thread not started. args deleted, Chunk remains in queue.)" <<
358  " transfer_thread_counter: " << transfer_thread_counter <<
359  " futures.size(): " << futures.size() << endl);
360  }
361  }
362  }
363  else {
364  // No more Chunks and no futures means we're done here.
365  if(futures.empty())
366  done = true;
367  }
368  future_finished = false;
369  }
370  }
371  catch (...) {
372  // Complete all of the futures, otherwise we'll have threads out there using up resources
373  while(!futures.empty()){
374  if(futures.back().valid())
375  futures.back().get();
376  futures.pop_back();
377  }
378  // re-throw the exception
379  throw;
380  }
381 }

488 static unsigned long long
489 get_index(const vector<unsigned long long> &address_in_target, const vector<unsigned long long> &target_shape)
490 {
491  assert(address_in_target.size() == target_shape.size()); // ranks must be equal
492 
493  auto shape_index = target_shape.rbegin();
494  auto index = address_in_target.rbegin(), index_end = address_in_target.rend();
495 
496  unsigned long long multiplier_var = *shape_index++;
497  unsigned long long offset = *index++;
498 
499  while (index != index_end) {
500  assert(*index < *shape_index); // index < shape for each dim
501 
502  offset += multiplier_var * *index++;
503  multiplier_var *= *shape_index++;
504  }
505 
506  return offset;
507 }

666 void DmrppArray::insert_constrained_contiguous(Dim_iter dim_iter, unsigned long *target_index,
667  vector<unsigned long long> &subset_addr,
668  const vector<unsigned long long> &array_shape, char /*Chunk*/*src_buf)
669 {
670  BESDEBUG("dmrpp", "DmrppArray::" << __func__ << "() - subsetAddress.size(): " << subset_addr.size() << endl);
671 
672  unsigned int bytes_per_elem = prototype()->width();
673 
674  char *dest_buf = get_buf();
675 
676  unsigned int start = this->dimension_start(dim_iter, true);
677  unsigned int stop = this->dimension_stop(dim_iter, true);
678  unsigned int stride = this->dimension_stride(dim_iter, true);
679 
680  dim_iter++;
681 
682  // The end case for the recursion is dimIter == dim_end(); stride == 1 is an optimization
683  // See the else clause for the general case.
684  if (dim_iter == dim_end() && stride == 1) {
685  // For the start and stop indexes of the subset, get the matching indexes in the whole array.
686  subset_addr.push_back(start);
687  unsigned long long start_index = get_index(subset_addr, array_shape);
688  subset_addr.pop_back();
689 
690  subset_addr.push_back(stop);
691  unsigned long long stop_index = get_index(subset_addr, array_shape);
692  subset_addr.pop_back();
693 
694  // Copy data block from start_index to stop_index
695  // TODO Replace this loop with a call to std::memcpy()
696  for (unsigned long source_index = start_index; source_index <= stop_index; source_index++) {
697  unsigned long target_byte = *target_index * bytes_per_elem;
698  unsigned long source_byte = source_index * bytes_per_elem;
699  // Copy a single value.
700  for (unsigned long i = 0; i < bytes_per_elem; i++) {
701  dest_buf[target_byte++] = src_buf[source_byte++];
702  }
703  (*target_index)++;
704  }
705 
706  }
707  else {
708  for (unsigned int myDimIndex = start; myDimIndex <= stop; myDimIndex += stride) {
709 
710  // Is it the last dimension?
711  if (dim_iter != dim_end()) {
712  // Nope! Then we recurse to the last dimension to read stuff
713  subset_addr.push_back(myDimIndex);
714  insert_constrained_contiguous(dim_iter, target_index, subset_addr, array_shape, src_buf);
715  subset_addr.pop_back();
716  }
717  else {
718  // We are at the last (inner most) dimension, so it's time to copy values.
719  subset_addr.push_back(myDimIndex);
720  unsigned int sourceIndex = get_index(subset_addr, array_shape);
721  subset_addr.pop_back();
722 
723  // Copy a single value.
724  unsigned long target_byte = *target_index * bytes_per_elem;
725  unsigned long source_byte = sourceIndex * bytes_per_elem;
726 
727  for (unsigned int i = 0; i < bytes_per_elem; i++) {
728  dest_buf[target_byte++] = src_buf[source_byte++];
729  }
730  (*target_index)++;
731  }
732  }
733  }
734 }

752 void DmrppArray::read_contiguous()
753 {
754  BESStopWatch sw;
755  if (BESDebug::IsSet(TIMING_LOG_KEY)) sw.start(prolog + " name: "+name(), "");
756 
757  // Get the single chunk that makes up this CONTIGUOUS variable.
758  auto chunk_refs = get_chunks();
759  if (chunk_refs.size() != 1)
760  throw BESInternalError(string("Expected only a single chunk for variable ") + name(), __FILE__, __LINE__);
761 
762  // This is the original chunk for this 'contiguous' variable.
763  auto the_one_chunk = chunk_refs[0];
764 
765  unsigned long long the_one_chunk_offset = the_one_chunk->get_offset();
766  unsigned long long the_one_chunk_size = the_one_chunk->get_size();
767 
768  // We only want to read in the Chunk concurrently if:
769  // - Concurrent transfers are enabled (DmrppRequestHandler::d_use_transfer_threads)
770  // - The variables size is above the threshold value held in DmrppRequestHandler::d_contiguous_concurrent_threshold
771  if (!DmrppRequestHandler::d_use_transfer_threads || the_one_chunk_size <= DmrppRequestHandler::d_contiguous_concurrent_threshold) {
772  // Read the the_one_chunk as is. This is the non-parallel I/O case
773  the_one_chunk->read_chunk();
774  }
775  else {
776 
777  // Allocate memory for the 'the_one_chunk' so the transfer threads can transfer data
778  // from the child chunks to it.
779  the_one_chunk->set_rbuf_to_size();
780

782  // If the size of the the_one_chunk is 3MB then 3 chunks will be made. We will round down
783  // when necessary and handle the remainder later on (3.3MB = 3 chunks, 4.2MB = 4 chunks, etc.)
784  unsigned long long num_chunks = floor(the_one_chunk_size / MB);
785  if (num_chunks >= DmrppRequestHandler::d_max_transfer_threads)
786  num_chunks = DmrppRequestHandler::d_max_transfer_threads;
787 
788  // Use the original chunk's size and offset to evenly split it into smaller chunks
789  unsigned long long chunk_size = the_one_chunk_size / num_chunks;
790  std::string chunk_byteorder = the_one_chunk->get_byte_order();
791 
792  // If the size of the the_one_chunk is not evenly divisible by num_chunks, capture
793  // the remainder here and increase the size of the last chunk by this number of bytes.
794  unsigned long long chunk_remainder = the_one_chunk_size % num_chunks;
795 
796  string chunk_url = the_one_chunk->get_data_url();
797 
798  // Setup a queue to break up the original the_one_chunk and keep track of the pieces
799  queue<shared_ptr<Chunk>> chunks_to_read;
800

802  unsigned long long chunk_offset = the_one_chunk_offset;
803  for (unsigned int i = 0; i < num_chunks - 1; i++) {
804  chunks_to_read.push(shared_ptr<Chunk>(new Chunk(chunk_url, chunk_byteorder, chunk_size, chunk_offset)));
805  chunk_offset += chunk_size;
806  }
807  // Make the the remainder Chunk, see above for details.
808  chunks_to_read.push(shared_ptr<Chunk>(new Chunk(chunk_url, chunk_byteorder, chunk_size + chunk_remainder, chunk_offset)));
809 
810  // We maintain a list of futures to track our parallel activities.
811  list<future<bool>> futures;
812  try {
813  bool done = false;
814  bool future_finished = true;
815  while (!done) {
816 
817  if (!futures.empty())
818  future_finished = get_next_future(futures, transfer_thread_counter, DMRPP_WAIT_FOR_FUTURE_MS, prolog);
819 
820  // If future_finished is true this means that the chunk_processing_thread_counter has been decremented,
821  // because future::get() was called or a call to future::valid() returned false.
822  BESDEBUG(dmrpp_3, prolog << "future_finished: " << (future_finished ? "true" : "false") << endl);
823 
824  if (!chunks_to_read.empty()) {
825  // Next we try to add a new Chunk compute thread if we can - there might be room.
826  bool thread_started = true;
827  while (thread_started && !chunks_to_read.empty()) {
828  auto current_chunk = chunks_to_read.front();
829  BESDEBUG(dmrpp_3, prolog << "Starting thread for " << current_chunk->to_string() << endl);
830 
831  auto args = unique_ptr<one_child_chunk_args_new>(new one_child_chunk_args_new(current_chunk, the_one_chunk));
832  thread_started = start_one_child_chunk_thread(futures, std::move(args));
833 
834  if (thread_started) {
835  chunks_to_read.pop();
836  BESDEBUG(dmrpp_3, prolog << "STARTED thread for " << current_chunk->to_string() << endl);
837  } else {
838  // Thread did not start, ownership of the arguments was not passed to the thread.
839  BESDEBUG(dmrpp_3, prolog << "Thread not started. args deleted, Chunk remains in queue.)" <<
840  " transfer_thread_counter: " << transfer_thread_counter <<
841  " futures.size(): " << futures.size() << endl);
842  }
843  }
844  } else {
845  // No more Chunks and no futures means we're done here.
846  if (futures.empty())
847  done = true;
848  }
849  future_finished = false;
850  }
851  }
852  catch (...) {
853  // Complete all of the futures, otherwise we'll have threads out there using up resources
854  while (!futures.empty()) {
855  if (futures.back().valid())
856  futures.back().get();
857  futures.pop_back();
858  }
859  // re-throw the exception
860  throw;
861  }
862  }
863 
864  // Now that the_one_chunk has been read, we do the needful...
865  if (is_deflate_compression() || is_shuffle_compression())
866  the_one_chunk->inflate_chunk(is_deflate_compression(), is_shuffle_compression(), get_chunk_size_in_elements(),var()->width());
867 
868  // The 'the_one_chunk' now holds the data values. Transfer it to the Array.
869  if (!is_projected()) { // if there is no projection constraint
870  reserve_value_capacity(get_size(false));
871  val2buf(the_one_chunk->get_rbuf()); // yes, it's not type-safe
872  }
873  else { // apply the constraint
874  vector<unsigned long long> array_shape = get_shape(false);
875

877  reserve_value_capacity(get_size(true));

879  vector<unsigned long long> subset;
