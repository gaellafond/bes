238 void process_chunks_concurrent(
239  const string &super_chunk_id,
240  queue<shared_ptr<Chunk>> &chunks,
241  DmrppArray *array,
242  const vector<unsigned long long> &constrained_array_shape ){
243 
244  // We maintain a list of futures to track our parallel activities.
245  list<future<bool>> futures;
246  try {
247  bool done = false;
248  bool future_finished = true;
249  while (!done) {
250 
251  if(!futures.empty())
252  future_finished = get_next_future(futures, chunk_processing_thread_counter, DMRPP_WAIT_FOR_FUTURE_MS, prolog);
253 
254  // If future_finished is true this means that the chunk_processing_thread_counter has been decremented,
255  // because future::get() was called or a call to future::valid() returned false.
256  BESDEBUG(SUPER_CHUNK_MODULE, prolog << "future_finished: " << (future_finished ? "true" : "false") << endl);
257 
258  if (!chunks.empty()){
259  // Next we try to add a new Chunk compute thread if we can - there might be room.
260  bool thread_started = true;
261  while(thread_started && !chunks.empty()) {
262  auto chunk = chunks.front();
263  BESDEBUG(SUPER_CHUNK_MODULE, prolog << "Starting thread for " << chunk->to_string() << endl);
264 
265  auto args = unique_ptr<one_chunk_args>(new one_chunk_args(super_chunk_id, chunk, array, constrained_array_shape));
266  thread_started = start_one_chunk_compute_thread(futures, std::move(args));
267 
268  if (thread_started) {
269  chunks.pop();
270  BESDEBUG(SUPER_CHUNK_MODULE, prolog << "STARTED thread for " << chunk->to_string() << endl);
271  } else {
272  // Thread did not start, ownership of the arguments was not passed to the thread.
273  BESDEBUG(SUPER_CHUNK_MODULE, prolog << "Thread not started. args deleted, Chunk remains in queue.) " <<
274  "chunk_processing_thread_counter: " << chunk_processing_thread_counter << " futures.size(): " << futures.size() << endl);
275  }
276  }
277  }
278  else {
279  // No more Chunks and no futures means we're done here.
280  if(futures.empty())
281  done = true;
282  }
283  future_finished = false;
284  }
285  }
286  catch (...) {
287  // Complete all of the futures, otherwise we'll have threads out there using up resources
288  while(!futures.empty()){
289  if(futures.back().valid())
290  futures.back().get();
291  futures.pop_back();
292  }
293  // re-throw the exception
294  throw;
295  }
296 }

402 bool SuperChunk::add_chunk(const std::shared_ptr<Chunk> candidate_chunk) {
403  bool chunk_was_added = false;

415  }

470 void SuperChunk::read_aggregate_bytes()
471 {
472  // Since we already have a good infrastructure for reading Chunks, we just make a big-ol-Chunk to
473  // use for grabbing bytes. Then, once read, we'll use the child Chunks to do the dirty work of inflating
474  // and moving the results into the DmrppCommon object.
475  Chunk chunk(d_data_url, "NOT_USED", d_size, d_offset);
476 
477  chunk.set_read_buffer(d_read_buffer, d_size,0,false);
478 
479  dmrpp_easy_handle *handle = DmrppRequestHandler::curl_handle_pool->get_easy_handle(&chunk);
480  if (!handle)
481  throw BESInternalError(prolog + "No more libcurl handles.", __FILE__, __LINE__);
482 
483  try {
484  handle->read_data(); // throws if error
485  DmrppRequestHandler::curl_handle_pool->release_handle(handle);
486  }
487  catch(...) {
488  DmrppRequestHandler::curl_handle_pool->release_handle(handle);
489  throw;
490  }
491 
492  // If the expected byte count was not read, it's an error.
493  if (d_size != chunk.get_bytes_read()) {
494  ostringstream oss;
495  oss << "Wrong number of bytes read for chunk; read: " << chunk.get_bytes_read() << ", expected: " << d_size;
496  throw BESInternalError(oss.str(), __FILE__, __LINE__);
497  }
498  d_is_read = true;
499 }

505 void SuperChunk::retrieve_data() {
506  if (d_is_read) {
507  BESDEBUG(SUPER_CHUNK_MODULE, prolog << "SuperChunk (" << (void **) this << ") has already been read! Returning." << endl);
508  return;
509  }
510 
511  if(!d_read_buffer){
512  // Allocate memory for SuperChunk receive buffer.
513  // release memory in destructor.
514  d_read_buffer = new char[d_size];
515  }
516 
517  // Massage the chunks so that their read/receive/intern data buffer
518  // points to the correct section of the d_read_buffer memory.
519  // "Slice it up!"
520  map_chunks_to_buffer();
521 
522  // Read the bytes from the target URL. (pthreads, maybe depends on size...)
523  // Use one (or possibly more) thread(s) depending on d_size
524  // and utilize our friend cURL to stuff the bytes into d_read_buffer
525  read_aggregate_bytes();
526 
527  // Set each Chunk's read state to true.
528  // Set each chunks byte count to the expected
529  // size for the chunk - because upstream events
530  // have assured this to be true.
531  for(auto chunk : d_chunks){
532  chunk->set_is_read(true);
533  chunk->set_bytes_read(chunk->get_size());
534  }
535 }
