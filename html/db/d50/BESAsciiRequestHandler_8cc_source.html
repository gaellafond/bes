64 bool BESAsciiRequestHandler::dap_build_version(BESDataHandlerInterface &dhi)
65 {
66  BESResponseObject *response = dhi.response_handler->get_response_object();
67  BESVersionInfo *info = dynamic_cast<BESVersionInfo *>(response);
68  if (!info)
69  throw BESInternalError("cast error", __FILE__, __LINE__);
70 
71  info->add_module(MODULE_NAME, MODULE_VERSION);
72 
73  return true;
74 }
