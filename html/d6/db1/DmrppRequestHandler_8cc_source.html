285 
286  delete bdmr->get_dmr();
287  bdmr->set_dmr(dmrpp);
288  }
289  else {
290  build_dmr_from_file(dhi.container, bdmr->get_dmr());
291  }
292 
293  bdmr->set_dap4_constraint(dhi);
294  bdmr->set_dap4_function(dhi);
295  }
296  catch (BESError &e) {
297  throw e;
298  }
299  catch (InternalErr & e) {
300  throw BESDapError(e.get_error_message(), true, e.get_error_code(), __FILE__, __LINE__);
301  }

306  throw BESInternalFatalError("Unknown exception caught building DAP4 Data response", __FILE__, __LINE__);

317 bool DmrppRequestHandler::dap_build_dap2data(BESDataHandlerInterface & dhi)
318 {
319  BESStopWatch sw;
320  if (BESDebug::IsSet(TIMING_LOG_KEY)) sw.start(prolog + "timer" , dhi.data[REQUEST_ID]);
321 
322  BESDEBUG(MODULE, prolog << "BEGIN" << endl);
323 
324  BESResponseObject *response = dhi.response_handler->get_response_object();
325  BESDataDDSResponse *bdds = dynamic_cast<BESDataDDSResponse *>(response);
326  if (!bdds) throw BESInternalError("Cast error, expected a BESDataDDSResponse object.", __FILE__, __LINE__);
327 
328  try {
329  string container_name_str = bdds->get_explicit_containers() ? dhi.container->get_symbolic_name() : "";
330 
331  DDS *dds = bdds->get_dds();
332  if (!container_name_str.empty()) dds->container_name(container_name_str);
333  string accessed = dhi.container->access();
334 
335  // Look in memory cache, if it's initialized
336  DDS *cached_dds_ptr = 0;
337  if (dds_cache && (cached_dds_ptr = static_cast<DDS*>(dds_cache->get(accessed)))) {
338  // copy the cached DAS into the BES response object
339  BESDEBUG(MODULE, prolog << "DDS Cached hit for : " << accessed << endl);
340  *dds = *cached_dds_ptr;
341  bdds->set_constraint(dhi);
342  }
343  else {
344  // Not in the local binary cache, make one...
345  DMR *dmr = NULL;
346 
347  // Check the Container to see if the handler should get the response from the MDS.
348  if (dhi.container->get_attributes().find(MDS_HAS_DMRPP) != string::npos) {
349  DmrppMetadataStore *mds = DmrppMetadataStore::get_instance();
350  if (!mds)
351  throw BESInternalError("MDS configuration error: The DMR++ module could not find the MDS", __FILE__, __LINE__);
352 
353  dmr = mds->get_dmrpp_object(dhi.container->get_relative_name());
354  if (!dmr)
355  throw BESInternalError("DMR++ module error: Null DMR++ object read from the MDS", __FILE__, __LINE__);
356  }
357  else {
358  dmr = new DMR();
359  build_dmr_from_file(dhi.container, dmr);
360  }
361 
362  // delete the current one;
363  delete dds;
364  // assign the new one.
365  dds = dmr->getDDS();
366 
367  // Stuff it into the response.
368  bdds->set_dds(dds);
369  bdds->set_constraint(dhi);
370 
371  delete dmr;
372 
373  // Cache it, if the cache is active.
374  if (dds_cache) {
375  dds_cache->add(new DDS(*dds), accessed);
376  }
377  }
378 
379  bdds->clear_container();
380  }
381  catch (BESError &e) {
382  throw;
383  }
384  catch (InternalErr & e) {
385  BESDapError ex(e.get_error_message(), true, e.get_error_code(), __FILE__, __LINE__);
386  throw ex;
387  }
388  catch (Error & e) {
389  BESDapError ex(e.get_error_message(), false, e.get_error_code(), __FILE__, __LINE__);
390  throw ex;
391  }
392  catch (std::exception &e) {
393  string s = string("C++ Exception: ") + e.what();
394  BESInternalFatalError ex(s, __FILE__, __LINE__);
395  throw ex;
396  }
397  catch (...) {
398  string s = "unknown exception caught building DDS";
399  BESInternalFatalError ex(s, __FILE__, __LINE__);
400  throw ex;
401  }
402 
403  BESDEBUG(MODULE, prolog << "END" << endl);
404  return true;

489 bool DmrppRequestHandler::dap_build_das(BESDataHandlerInterface & dhi)
490 {
491  BESStopWatch sw;
492  if (BESDebug::IsSet(TIMING_LOG_KEY)) sw.start(prolog + "timer" , dhi.data[REQUEST_ID]);
493 
494  BESResponseObject *response = dhi.response_handler->get_response_object();
495  BESDASResponse *bdas = dynamic_cast<BESDASResponse *>(response);
496  if (!bdas) throw BESInternalError("Cast error, expected a BESDASResponse object.", __FILE__, __LINE__);
497 
498  try {
499  string container_name_str = bdas->get_explicit_containers() ? dhi.container->get_symbolic_name() : "";
500 
501  DAS *das = bdas->get_das();
502  if (!container_name_str.empty()) das->container_name(container_name_str);
503  string accessed = dhi.container->access();
504 
505  // Look in memory cache (if it's initialized)
506  DAS *cached_das_ptr = 0;
507  if (das_cache && (cached_das_ptr = static_cast<DAS*>(das_cache->get(accessed)))) {
508  // copy the cached DAS into the BES response object
509  *das = *cached_das_ptr;
510  }
511  else {
512  // Not in cache, better make one!
513  // 1) Build a DMR
514  DMR *dmr = new DMR();
515  build_dmr_from_file(dhi.container, dmr);
516 
517  // Get a DDS from the DMR
518  DDS *dds = dmr->getDDS();
519 
520  // Load the BESDASResponse DAS from the DDS
521  dds->get_das(das);
522 
523  delete dds;
524  delete dmr;
525 
526  Ancillary::read_ancillary_das(*das, accessed);
527  // Add to cache if cache is active
528  if (das_cache) {
529  das_cache->add(new DAS(*das), accessed);
530  }
531  }
532 
533  bdas->clear_container();
534  }
535  catch (BESError &e) {
536  throw;
537  }
538  catch (InternalErr & e) {
539  BESDapError ex(e.get_error_message(), true, e.get_error_code(), __FILE__, __LINE__);
540  throw ex;
541  }
542  catch (Error & e) {
543  BESDapError ex(e.get_error_message(), false, e.get_error_code(), __FILE__, __LINE__);
544  throw ex;
545  }
546  catch (std::exception &e) {
547  string s = string("C++ Exception: ") + e.what();
548  BESInternalFatalError ex(s, __FILE__, __LINE__);
549  throw ex;
550  }
551  catch (...) {
552  string s = "unknown exception caught building DAS";
