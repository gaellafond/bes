76 void BESDefineResponseHandler::execute(BESDataHandlerInterface &dhi)
77 {
78  dhi.action_name = DEFINE_RESPONSE_STR;
79 
80  string def_name = dhi.data[DEF_NAME];
81  string store_name = dhi.data[STORE_NAME];
82 
83  BESDefinitionStorage *store = BESDefinitionStorageList::TheList()->find_persistence(store_name);
84  if (store) {
85  store->del_definition(def_name);
86 
87  BESDefine *dd = new BESDefine;
88  dhi.first_container();
89  while (dhi.container) {
90  dd->add_container(dhi.container);
91  dhi.next_container();
92  }
93 
94  store->add_definition(def_name, dd);
95  }
96  else {
97  throw BESSyntaxUserError(string("Unable to add definition '") + def_name + "' to '" + store_name
98  + "' store. Store does not exist", __FILE__, __LINE__);
99  }
100 }
