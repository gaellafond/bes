74 void BESVersionResponseHandler::execute(BESDataHandlerInterface &dhi)
75 {
76  BESVersionInfo *info = new BESVersionInfo();
77  d_response_object = info;
78  dhi.action_name = VERS_RESPONSE_STR;
79  info->begin_response( VERS_RESPONSE_STR, dhi);
80 
81  string admin_email = "";
82  try {
83  bes::ServerAdministrator sd;
84  admin_email = sd.get_email();
85  }
86  catch (...) {
87  admin_email = DEFAULT_ADMINISTRATOR;
88  }
89  if (admin_email.empty()) {
90  admin_email = DEFAULT_ADMINISTRATOR;
91  }
92  info->add_tag("Administrator", admin_email);
93 
94  info->add_library( PACKAGE_NAME, PACKAGE_VERSION);
95 
96  BESRequestHandlerList::TheList()->execute_all(dhi);
97 
98  info->end_response();
99 }
