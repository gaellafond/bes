56 void BESUncompress3GZ::uncompress(const string &src, int dest_fd)
57 {
58  // buffer to hold the uncompressed data
59  char in[CHUNK];
60 
61  // open the file to be read by gzopen. If the file is not compressed
62  // using gzip then all this function will do is trasnfer the data to the
63  // destination file.
64  gzFile gsrc = gzopen(src.c_str(), "rb");
65  if (gsrc == NULL) {
66  string err = "Could not open the compressed file " + src;
67  throw BESInternalError(err, __FILE__, __LINE__);
68  }
69 
70  // gzread will read the data in uncompressed. All we have to do is write
71  // it to the destination file.
72  bool done = false;
73  while (!done) {
74  int bytes_read = gzread(gsrc, in, CHUNK);
75  if (bytes_read == 0) {
76  done = true;
77  }
78  else {
79  int bytes_written = write(dest_fd, in, bytes_read);
80  if (bytes_written < bytes_read) {
81  ostringstream strm;
82  strm << "Error writing uncompressed data for file " << gsrc << ": wrote "
83  << bytes_written << " instead of " << bytes_read;
84  gzclose(gsrc);
85 
86  throw BESInternalError(strm.str(), __FILE__, __LINE__);
87  }
88  }
89  }
90 
91  gzclose(gsrc);
92 }
